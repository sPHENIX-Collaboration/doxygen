22 G4ParticleDefinition* ActsFatras::PDGtoG4Converter::getParticleDefinition(
23  Acts::PdgParticle pdgCode) const {
24  std::unordered_map<Acts::PdgParticle, G4ParticleDefinition*>::const_iterator
25  it = m_pdgG4ParticleMap.find(pdgCode);
26 
27  // Find the corresponding particle and return it
28  if (it != m_pdgG4ParticleMap.end()) {
29  return it->second;
30  } else {
31  // Rescue mechanism: If the particle is neutral and its anti-particle is
32  // stored then return that one instead
33  it = m_pdgG4ParticleMap.find(makeAbsolutePdgParticle(pdgCode));
34  if (it != m_pdgG4ParticleMap.end() &&
35  std::abs(it->second->GetPDGCharge()) < 0.1) {
36  return it->second;
37  }
38  }
39  // No particle found
40  return nullptr;
41 }
