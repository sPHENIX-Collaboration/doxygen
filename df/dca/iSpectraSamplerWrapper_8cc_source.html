119 void iSpectraSamplerWrapper::PassHadronListToJetscape() {
120  unsigned int nev = iSpectraSampler_ptr_->get_number_of_sampled_events();
121  VERBOSE(2) << "Passing all sampled hadrons to the JETSCAPE framework";
122  VERBOSE(4) << "number of events to pass : " << nev;
123  for (unsigned int iev = 0; iev < nev; iev++) {
124  std::vector<shared_ptr<Hadron>> hadrons;
125  unsigned int nparticles = (iSpectraSampler_ptr_->get_number_of_particles(iev));
126  VERBOSE(4) << "event " << iev << ": number of particles = "<< nparticles;
127  for (unsigned int ipart = 0; ipart < nparticles; ipart++) {
128  iSS_Hadron current_hadron = (iSpectraSampler_ptr_->get_hadron(iev, ipart));
129  int hadron_label = 0;
130  int hadron_status = -1;
131  int hadron_id = current_hadron.pid;
132  //int hadron_id = 1; // just for testing need to be changed to the line above
133  double hadron_mass = current_hadron.mass;
134  FourVector hadron_p(current_hadron.px, current_hadron.py,
135  current_hadron.pz, current_hadron.E);
136  FourVector hadron_x(current_hadron.x, current_hadron.y,
137  current_hadron.z, current_hadron.t);
138 
139  // create a JETSCAPE Hadron
140  hadrons.push_back(make_shared<Hadron>(hadron_label,hadron_id,hadron_status,hadron_p,hadron_x, hadron_mass));
141  //Hadron* jetscape_hadron = new Hadron(hadron_label, hadron_id, hadron_status, hadron_p, hadron_x, hadron_mass);
142  //(*Hadron_list_)[iev]->push_back(*jetscape_hadron);
143  }
144  Hadron_list_.push_back(hadrons);
145  }
146  VERBOSE(4) << "JETSCAPE received " << Hadron_list_.size() << " events.";
147  for (unsigned int iev = 0; iev < Hadron_list_.size(); iev++) {
148  VERBOSE(4) << "In event " << iev << " JETSCAPE received " << Hadron_list_.at(iev).size() << " particles.";
149  }
150 
151 }

153 void iSpectraSamplerWrapper::WriteTask(weak_ptr<JetScapeWriter> w)
154 {
155  VERBOSE(4)<<"In iSpectraSamplerWrapper::WriteTask";
156  auto f = w.lock();
157  if ( !f ) return;
158 
159  f->WriteComment("JetScape module: "+GetId());
160  if(Hadron_list_.size()>0) {
161  f->WriteComment("Final State Bulk Hadrons");
162  for(unsigned int j=0; j<Hadron_list_.size(); j++){
163  vector<shared_ptr<Hadron>> hadVec = Hadron_list_.at(j);
164  for(unsigned int i=0; i<hadVec.size(); i++) {
165  f->WriteWhiteSpace("["+to_string(i)+"] H");
166  f->Write(hadVec.at(i));
