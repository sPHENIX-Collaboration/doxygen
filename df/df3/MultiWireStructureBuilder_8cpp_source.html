60  Acts::Experimental::InternalStructure construct(
61  const Acts::GeometryContext& gctx) const final {
62  if (not m_cfg.auxiliary.empty()) {
63  ACTS_DEBUG(m_cfg.auxiliary);
64  }
65 
66  Acts::Experimental::DetectorVolumeUpdator internalVolumeUpdator =
67  Acts::Experimental::tryNoVolumes();
68  // Create the indexed surfaces
69  auto internalSurfaces = m_cfg.iSurfaces;
70  Acts::Experimental::detail::IndexedSurfacesGenerator<
71  decltype(internalSurfaces), Acts::Experimental::MultiLayerSurfacesImpl>
72  isg{internalSurfaces,
73  {},
74  {m_cfg.binning[0u].binValue, m_cfg.binning[1u].binValue},
75  {m_cfg.binning[0u].expansion, m_cfg.binning[1u].expansion}};
76  Acts::Experimental::detail::CenterReferenceGenerator rGenerator;
77  Acts::Experimental::detail::GridAxisGenerators::EqBoundEqBound aGenerator{
78  {m_cfg.binning[0u].edges.front(), m_cfg.binning[0u].edges.back()},
79  m_cfg.binning[0u].edges.size() - 1,
80  {m_cfg.binning[1u].edges.front(), m_cfg.binning[1u].edges.back()},
81  m_cfg.binning[1u].edges.size() - 1};
82 
83  auto sfCandidatesUpdator = isg(gctx, aGenerator, rGenerator);
84 
85  return {internalSurfaces,
86  {},
87  std::move(sfCandidatesUpdator),
88  std::move(internalVolumeUpdator)};
89  }
