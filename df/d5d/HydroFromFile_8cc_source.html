232 void HydroFromFile::GetHydroInfo(
233  Jetscape::real t, Jetscape::real x, Jetscape::real y, Jetscape::real z,
234  std::unique_ptr<FluidCellInfo>& fluid_cell_info_ptr) {
235  if (hydro_status != FINISHED) {
236  WARN << "Hydro not run yet ...";
237  exit(-1);
238  }
239 
240  double t_local = static_cast<double>(t);
241  double x_local = static_cast<double>(x);
242  double y_local = static_cast<double>(y);
243  double z_local = static_cast<double>(z);
244 
245  // initialize the fluid cell pointer
246  fluidCell *temp_fluid_cell_ptr = new fluidCell;
247  if (hydro_type_ == 1) { // for OSU 2+1d hydro
248  double tau_local = sqrt(t*t - z*z);
249  double eta_local = 0.5*log((t + z)/(t - z + 1e-15));
250  if (std::isnan(tau_local)) { // check
251  WARN << "[Error]: HydroFromFile::GetHydroInfo(): "
252  << "tau is nan!";
253  WARN << "please check: t = " << t << ", z = " << z;
254  exit(1);
255  }
256 #ifdef USE_HDF5
257  hydroinfo_h5_ptr->getHydroinfo(tau_local, x_local, y_local,
258  temp_fluid_cell_ptr);
259  // compute the flow velocity in the lab frame
260  double u0_perp = (
261  1./sqrt(1. - temp_fluid_cell_ptr->vx*temp_fluid_cell_ptr->vx
262  - temp_fluid_cell_ptr->vy*temp_fluid_cell_ptr->vy));
263  double u0 = u0_perp*cosh(eta_local);
264  temp_fluid_cell_ptr->vx *= u0_perp/u0;
265  temp_fluid_cell_ptr->vy *= u0_perp/u0;
266  temp_fluid_cell_ptr->vz = z/(t + 1e-15);
267 #endif
268  } else if (hydro_type_ == 2 || hydro_type_ == 3 || hydro_type_ == 4) {
269  hydroinfo_MUSIC_ptr->getHydroValues(x_local, y_local, z_local, t_local,
270  temp_fluid_cell_ptr);
271  }
272 
273  // assign all the quantites to JETSCAPE output
274  // thermodyanmic quantities
275  fluid_cell_info_ptr = std::make_unique<FluidCellInfo>();
276  fluid_cell_info_ptr->energy_density = (
277  static_cast<real>(temp_fluid_cell_ptr->ed));
278  fluid_cell_info_ptr->entropy_density = (
279  static_cast<real>(temp_fluid_cell_ptr->sd));
280  fluid_cell_info_ptr->temperature = (
281  static_cast<real>(temp_fluid_cell_ptr->temperature));
282  fluid_cell_info_ptr->pressure = (
283  static_cast<real>(temp_fluid_cell_ptr->pressure));
284  // QGP fraction
285  double qgp_fraction_local = 1.0;
286  if (temp_fluid_cell_ptr->temperature < T_c_) {
287  qgp_fraction_local = 0.0;
288  }
289  fluid_cell_info_ptr->qgp_fraction = static_cast<real>(qgp_fraction_local);
290  // chemical potentials
291  fluid_cell_info_ptr->mu_B = 0.0;
292  fluid_cell_info_ptr->mu_C = 0.0;
293  fluid_cell_info_ptr->mu_S = 0.0;
294  // dynamical quantites
295  fluid_cell_info_ptr->vx = static_cast<real>(temp_fluid_cell_ptr->vx);
296  fluid_cell_info_ptr->vy = static_cast<real>(temp_fluid_cell_ptr->vy);
297  fluid_cell_info_ptr->vz = static_cast<real>(temp_fluid_cell_ptr->vz);
298  for (int i = 0; i < 4; i++) {
299  for (int j = 0; j < 4; j++) {
300  fluid_cell_info_ptr->pi[i][j] = (
301  static_cast<real>(temp_fluid_cell_ptr->pi[i][j]));
302  }
303  }
304  fluid_cell_info_ptr->bulk_Pi = (
305  static_cast<real>(temp_fluid_cell_ptr->bulkPi));
306  delete temp_fluid_cell_ptr;
307 }
