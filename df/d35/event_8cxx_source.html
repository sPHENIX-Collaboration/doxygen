18 constexpr double TINY = 1e-12;

20 // Generalized mean for p > 0.
21 // M_p(a, b) = (1/2*(a^p + b^p))^(1/p)
22 inline double positive_pmean(double p, double a, double b) {
23  return std::pow(.5*(std::pow(a, p) + std::pow(b, p)), 1./p);
24 }
25 
26 // Generalized mean for p < 0.
27 // Same as the positive version, except prevents division by zero.
28 inline double negative_pmean(double p, double a, double b) {
29  if (a < TINY || b < TINY)
30  return 0.;
31  return positive_pmean(p, a, b);
32 }
33 
34 // Generalized mean for p == 0.
35 inline double geometric_mean(double a, double b) {
36  return std::sqrt(a*b);
37 }

41 // Determine the grid parameters like so:
42 // 1. Read and set step size from the configuration.
43 // 2. Read grid max from the config, then set the number of steps as
44 // nsteps = ceil(2*max/step).
45 // 3. Set the actual grid max as max = nsteps*step/2. Hence if the step size
46 // does not evenly divide the config max, the actual max will be marginally
47 // larger (by at most one step size).
48 Event::Event(const VarMap& var_map)
49  : norm_(var_map["normalization"].as<double>()),
50  dxy_(var_map["grid-step"].as<double>()),
51  nsteps_(std::ceil(2.*var_map["grid-max"].as<double>()/dxy_)),
52  xymax_(.5*nsteps_*dxy_),
53  TA_(boost::extents[nsteps_][nsteps_]),
54  TB_(boost::extents[nsteps_][nsteps_]),
55  TR_(boost::extents[nsteps_][nsteps_]) {
56  // Choose which version of the generalized mean to use based on the
57  // configuration. The possibilities are defined above. See the header for
58  // more information.
59  auto p = var_map["reduced-thickness"].as<double>();
60 
61  if (std::fabs(p) < TINY) {
62  compute_reduced_thickness_ = [this]() {
63  compute_reduced_thickness(geometric_mean);
64  };
65  } else if (p > 0.) {
66  compute_reduced_thickness_ = [this, p]() {
67  compute_reduced_thickness(
68  [p](double a, double b) { return positive_pmean(p, a, b); });
69  };
70  } else {
71  compute_reduced_thickness_ = [this, p]() {
72  compute_reduced_thickness(
73  [p](double a, double b) { return negative_pmean(p, a, b); });
74  };
75  }
76 }

103 void Event::compute_nuclear_thickness(
104  const Nucleus& nucleus, NucleonProfile& profile, Grid& TX) {
105  // Construct the thickness grid by looping over participants and adding each
106  // to a small subgrid within its radius. Compared to the other possibility
107  // (grid cells as the outer loop and participants as the inner loop), this
108  // reduces the number of required distance-squared calculations by a factor of
109  // ~20 (depending on the nucleon size). The Event unit test verifies that the
110  // two methods agree.
111 
112  // Wipe grid with zeros.
113  std::fill(TX.origin(), TX.origin() + TX.num_elements(), 0.);
114 
115  const double r = profile.radius();
116 
117  // Deposit each participant onto the grid.
118  for (const auto& nucleon : nucleus) {
119  if (!nucleon.is_participant())
120  continue;
121 
122  ++npart_;
123 
124  // Work in coordinates relative to (-width/2, -width/2).
125  double x = nucleon.x() + xymax_;
126  double y = nucleon.y() + xymax_;
127 
128  // Determine min & max indices of nucleon subgrid.
129  int ixmin = clip(static_cast<int>((x-r)/dxy_), 0, nsteps_-1);
130  int iymin = clip(static_cast<int>((y-r)/dxy_), 0, nsteps_-1);
131  int ixmax = clip(static_cast<int>((x+r)/dxy_), 0, nsteps_-1);
132  int iymax = clip(static_cast<int>((y+r)/dxy_), 0, nsteps_-1);
133 
134  // Prepare profile for new nucleon.
135  profile.fluctuate();
136 
137  // Add profile to grid.
138  for (auto iy = iymin; iy <= iymax; ++iy) {
139  double dysq = std::pow(y - (static_cast<double>(iy)+.5)*dxy_, 2);
140  for (auto ix = ixmin; ix <= ixmax; ++ix) {
141  double dxsq = std::pow(x - (static_cast<double>(ix)+.5)*dxy_, 2);
142  TX[iy][ix] += profile.thickness(dxsq + dysq);
143  }
144  }
145  }
146 }

148 template <typename GenMean>
149 void Event::compute_reduced_thickness(GenMean gen_mean) {
150  double sum = 0.;
151  double ixcm = 0.;
152  double iycm = 0.;
153 
154  for (int iy = 0; iy < nsteps_; ++iy) {
155  for (int ix = 0; ix < nsteps_; ++ix) {
156  auto t = norm_ * gen_mean(TA_[iy][ix], TB_[iy][ix]);
157  TR_[iy][ix] = t;
158  sum += t;
159  // Center of mass grid indices.
160  // No need to multiply by dxy since it would be canceled later.
161  ixcm += t * static_cast<double>(ix);
162  iycm += t * static_cast<double>(iy);
163  }
164  }
165 
166  multiplicity_ = dxy_ * dxy_ * sum;
167  ixcm_ = ixcm / sum;
168  iycm_ = iycm / sum;
169 }

171 void Event::compute_observables() {
172  // Compute eccentricity.
173 
174  // Simple helper class for use in the following loop.
175  struct EccentricityAccumulator {
176  double re = 0.; // real part
177  double im = 0.; // imaginary part
178  double wt = 0.; // weight
179  double ecc() const // compute final eccentricity
180  { return std::sqrt(re*re + im*im) / std::fmax(wt, TINY); }
181 
182  double phi() const // nth participant plane angle, n*phi_n
183  { return std::atan2(im, re); }
184  } e2, e3, e4, e5;
185 
186  for (int iy = 0; iy < nsteps_; ++iy) {
187  for (int ix = 0; ix < nsteps_; ++ix) {
188  const auto& t = TR_[iy][ix];
189  if (t < TINY)
190  continue;
191 
192  // Compute (x, y) relative to the CM and cache powers of x, y, r.
193  auto x = static_cast<double>(ix) - ixcm_;
194  auto x2 = x*x;
195  auto x3 = x2*x;
196  auto x4 = x2*x2;
197 
198  auto y = static_cast<double>(iy) - iycm_;
199  auto y2 = y*y;
200  auto y3 = y2*y;
201  auto y4 = y2*y2;
202 
203  auto r2 = x2 + y2;
204  auto r = std::sqrt(r2);
205  auto r4 = r2*r2;
206 
207  auto xy = x*y;
208  auto x2y2 = x2*y2;
209 
210  // The eccentricity harmonics are weighted averages of r^n*exp(i*n*phi)
211  // over the entropy profile (reduced thickness). The naive way to compute
212  // exp(i*n*phi) at a given (x, y) point is essentially:
213  //
214  // phi = arctan2(y, x)
215  // real = cos(n*phi)
216  // imag = sin(n*phi)
217  //
218  // However this implementation uses three unnecessary trig functions; a
219  // much faster method is to express the cos and sin directly in terms of x
220  // and y. For example, it is trivial to show (by drawing a triangle and
221  // using rudimentary trig) that
222  //
223  // cos(arctan2(y, x)) = x/r = x/sqrt(x^2 + y^2)
224  // sin(arctan2(y, x)) = y/r = x/sqrt(x^2 + y^2)
225  //
226  // This is easily generalized to cos and sin of (n*phi) by invoking the
227  // multiple angle formula, e.g. sin(2x) = 2sin(x)cos(x), and hence
228  //
229  // sin(2*arctan2(y, x)) = 2*sin(arctan2(y, x))*cos(arctan2(y, x))
230  // = 2*x*y / r^2
231  //
232  // Which not only eliminates the trig functions, but also naturally
233  // cancels the r^2 weight. This cancellation occurs for all n.
234  //
235  // The Event unit test verifies that the two methods agree.
236  e2.re += - t * (x2 - y2);
237  e2.im += t * 2.*xy;
238  e2.wt += t * r2;
239 
240  e3.re += - t * (x3 - 3.*x*y2);
241  e3.im += t * (3.*y*x2 - y3);
242  e3.wt += t * r2*r;
243 
244  e4.re += - t * (x4 + y4 - 6.*x2y2);
245  e4.im += t * 4.*xy*(x2 - y2);
246  e4.wt += t * r4;
247 
248  e5.re += - t * y*(5.*y4 - 10.*x2y2 + x4);
249  e5.im += t * y*(y4 - 10.*x2y2 + 5.*x4);
250  e5.wt += t * r4*r;
251  }
252  }
253 
254  eccentricity_[2] = e2.ecc();
255  eccentricity_[3] = e3.ecc();
256  eccentricity_[4] = e4.ecc();
257  eccentricity_[5] = e5.ecc();
258 
259  participant_plane_[2] = e2.phi() / 2.;
260  participant_plane_[3] = e3.phi() / 3.;
261  participant_plane_[4] = e4.phi() / 4.;
262  participant_plane_[5] = e5.phi() / 5.;
263 }
