36 void PHG4EPDetector::ConstructMe(G4LogicalVolume* world)
37 {
38  G4Material* material = G4Material::GetMaterial("G4_PLASTIC_SC_VINYLTOLUENE");
39 
40  G4VisAttributes* attrs = new G4VisAttributes();
41 
42  attrs->SetVisibility(true);
43  attrs->SetForceSolid(true);
44  attrs->SetColour(G4Colour::Red());
45 
46  G4ThreeVector positive(0., 0., m_z_position * cm);
47  G4ThreeVector negative(0., 0., -m_z_position * cm);
48 
49  constexpr int32_t ntiles = 31;
50  constexpr int32_t nslices = 12;
51 
52  for (int32_t i = 0; i < ntiles; ++i)
53  {
54  std::string label = "EPD_tile_" + std::to_string(i);
55 
56  G4ExtrudedSolid* block = construct_block(i);
57  G4LogicalVolume* volume = new G4LogicalVolume(
58  block, material, label.data(), 0, 0, 0);
59 
60  volume->SetVisAttributes(attrs);
61 
62  for (int32_t k = 0; k < nslices; ++k)
63  {
64  G4RotationMatrix* rotate = new G4RotationMatrix();
65 
66  rotate->rotateZ(k * 2 * M_PI / nslices);
67 
68  m_volumes.emplace(
69  new G4PVPlacement(
70  rotate, positive, volume, label, world,
71  false, 2 * k + 0, OverlapCheck()),
72  module_id_for(i, k, 0));
73 
74  m_volumes.emplace(
75  new G4PVPlacement(
76  rotate, negative, volume, label, world,
77  false, 2 * k + 1, OverlapCheck()),
78  module_id_for(i, k, 1));
79  }
80  }
81 }

135 G4ExtrudedSolid* PHG4EPDetector::construct_block(int32_t index)
136 {
137  G4String label("tile_" + std::to_string(index));
138 
139  const double(*coords)[6][2] = &coordinates[index];
140 
141  std::vector<G4TwoVector> vertices;
142 
143  for (int32_t i = 0; i < 6; ++i)
144  {
145  double x = (*coords)[i][0];
146  double y = (*coords)[i][1];
147 
148  if (x == 0. && y == 0.)
149  continue;
150 
151  vertices.emplace_back(x * mm, y * mm);
152  }
153 
154  std::vector<G4ExtrudedSolid::ZSection> zsections = {
155  G4ExtrudedSolid::ZSection(-dz, G4TwoVector(), 1.),
156  G4ExtrudedSolid::ZSection(dz, G4TwoVector(), 1.),
157  };
158 
159  return new G4ExtrudedSolid(label, vertices, zsections);
160 }
