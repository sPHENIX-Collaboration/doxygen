11 std::vector<ActsExamples::ParticleHitCount>
12 ActsExamples::SimMultiTrajectory::identifyMajorityParticle(
13  const size_t& entryIndex) const {
14  std::vector<ActsExamples::ParticleHitCount> particleHitCount;
15  particleHitCount.reserve(10);
16  if (not m_trackTips.empty()) {
17  if (not hasTrajectory(entryIndex)) {
18  return particleHitCount;
19  }
20  m_multiTrajectory.visitBackwards(entryIndex, [&](const auto& state) {
21  // No truth info with non-measurement state
22  if (not state.typeFlags().test(Acts::TrackStateFlag::MeasurementFlag)) {
23  return true;
24  }
25  // Find the truth particle associated with this state
26  const auto particleId = state.uncalibrated().truthHit().particleId();
27  // Find if the particle already exists
28  auto it = std::find_if(particleHitCount.begin(), particleHitCount.end(),
29  [=](const ActsExamples::ParticleHitCount& phc) {
30  return phc.particleId == particleId;
31  });
32 
33  // Either increase count if we saw the particle before or add it
34  if (it != particleHitCount.end()) {
35  it->hitCount += 1;
36  } else {
37  particleHitCount.push_back({particleId, 1u});
38  }
39  return true;
40  });
41  }
42  if (not particleHitCount.empty()) {
43  // sort by hit count, i.e. majority particle first
44  std::sort(particleHitCount.begin(), particleHitCount.end(),
45  [](const ActsExamples::ParticleHitCount& lhs,
46  const ActsExamples::ParticleHitCount& rhs) {
47  return lhs.hitCount > rhs.hitCount;
48  });
49  }
50 
51  return particleHitCount;
52 }
