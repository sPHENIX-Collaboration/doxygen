56 ActsExamples::ProcessCode ActsExamples::CsvParticleReader::read(
57  const ActsExamples::AlgorithmContext& ctx) {
58  SimParticleContainer::sequence_type unordered;
59 
60  auto path = perEventFilepath(m_cfg.inputDir, m_cfg.inputStem + ".csv",
61  ctx.eventNumber);
62  // vt and m are an optional columns
63  dfe::NamedTupleCsvReader<ParticleData> reader(path, {"vt", "m"});
64  ParticleData data;
65 
66  while (reader.read(data)) {
67  ActsFatras::Particle particle(ActsFatras::Barcode(data.particle_id),
68  Acts::PdgParticle(data.particle_type),
69  data.q * Acts::UnitConstants::e,
70  data.m * Acts::UnitConstants::GeV);
71  particle.setProcess(static_cast<ActsFatras::ProcessType>(data.process));
72  particle.setPosition4(
73  data.vx * Acts::UnitConstants::mm, data.vy * Acts::UnitConstants::mm,
74  data.vz * Acts::UnitConstants::mm, data.vt * Acts::UnitConstants::ns);
75  // Only used for direction; normalization/units do not matter
76  particle.setDirection(data.px, data.py, data.pz);
77  particle.setAbsoluteMomentum(std::hypot(data.px, data.py, data.pz) *
78  Acts::UnitConstants::GeV);
79  unordered.push_back(std::move(particle));
80  }
81 
82  // Write ordered particles container to the EventStore
83  SimParticleContainer particles;
84  particles.insert(unordered.begin(), unordered.end());
85  m_outputParticles(ctx, std::move(particles));
86 
87  return ProcessCode::SUCCESS;
88 }
