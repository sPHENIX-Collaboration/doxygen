246 void PHTpcClusterMover::CircleFitByTaubin (std::vector<TrkrCluster*> clusters, double &R, double &X0, double &Y0)
247 /*
248  Circle fit to a given set of data points (in 2D)
249  This is an algebraic fit, due to Taubin, based on the journal article
250  G. Taubin, "Estimation Of Planar Curves, Surfaces And Nonplanar
251  Space Curves Defined By Implicit Equations, With
252  Applications To Edge And Range Image Segmentation",
253  IEEE Trans. PAMI, Vol. 13, pages 1115-1138, (1991)
254  It works well whether data points are sampled along an entire circle or along a small arc.
255  It still has a small bias and its statistical accuracy is slightly lower than that of the geometric fit (minimizing geometric distances),
256  It provides a very good initial guess for a subsequent geometric fit.
257  Nikolai Chernov (September 2012)
258 */
259 {
260  int iter,IterMAX=99;
261 
262  double Mz,Mxy,Mxx,Myy,Mxz,Myz,Mzz,Cov_xy,Var_z;
263  double A0,A1,A2,A22,A3,A33;
264  double x,y;
265  double DET,Xcenter,Ycenter;
266 
267  // Compute x- and y- sample means
268  double meanX = 0;
269  double meanY = 0;
270  double weight = 0;
271  for(unsigned int iclus = 0; iclus < clusters.size(); ++iclus)
272  {
273  meanX += clusters[iclus]->getX();
274  meanY += clusters[iclus]->getY();
275  weight++;
276  }
277  meanX /= weight;
278  meanY /= weight;
279 
280  // computing moments
281 
282  Mxx=Myy=Mxy=Mxz=Myz=Mzz=0.;
283 
284  for (unsigned int i=0; i<clusters.size(); i++)
285  {
286  double Xi = clusters[i]->getX() - meanX; // centered x-coordinates
287  double Yi = clusters[i]->getY() - meanY; // centered y-coordinates
288  double Zi = Xi*Xi + Yi*Yi;
289 
290  Mxy += Xi*Yi;
291  Mxx += Xi*Xi;
292  Myy += Yi*Yi;
293  Mxz += Xi*Zi;
294  Myz += Yi*Zi;
295  Mzz += Zi*Zi;
296  }
297  Mxx /= weight;
298  Myy /= weight;
299  Mxy /= weight;
300  Mxz /= weight;
301  Myz /= weight;
302  Mzz /= weight;
303 
304  // computing coefficients of the characteristic polynomial
305 
306  Mz = Mxx + Myy;
307  Cov_xy = Mxx*Myy - Mxy*Mxy;
308  Var_z = Mzz - Mz*Mz;
309  A3 = 4*Mz;
310  A2 = -3*Mz*Mz - Mzz;
311  A1 = Var_z*Mz + 4*Cov_xy*Mz - Mxz*Mxz - Myz*Myz;
312  A0 = Mxz*(Mxz*Myy - Myz*Mxy) + Myz*(Myz*Mxx - Mxz*Mxy) - Var_z*Cov_xy;
313  A22 = A2 + A2;
314  A33 = A3 + A3 + A3;
315 
316  // finding the root of the characteristic polynomial
317  // using Newton's method starting at x=0
318  // (it is guaranteed to converge to the right root)
319 
320  for (x=0.,y=A0,iter=0; iter<IterMAX; iter++) // usually, 4-6 iterations are enough
321  {
322  double Dy = A1 + x*(A22 + A33*x);
323  double xnew = x - y/Dy;
324  if ((xnew == x)||(!std::isfinite(xnew))) break;
325  double ynew = A0 + xnew*(A1 + xnew*(A2 + xnew*A3));
326  if (fabs(ynew)>=fabs(y)) break;
327  x = xnew; y = ynew;
328  }
329 
330  // computing parameters of the fitting circle
331 
332  DET = x*x - x*Mz + Cov_xy;
333  Xcenter = (Mxz*(Myy - x) - Myz*Mxy)/DET/2;
334  Ycenter = (Myz*(Mxx - x) - Mxz*Mxy)/DET/2;
335 
336  // assembling the output
337 
338  X0 = Xcenter + meanX;
339  Y0 = Ycenter + meanY;
340  R = sqrt(Xcenter*Xcenter + Ycenter*Ycenter + Mz);
341 }

343 void PHTpcClusterMover::circle_circle_intersection(double r1, double r2, double x2, double y2, double &xplus, double &yplus, double &xminus, double &yminus)
344 {
345  // r1 is radius of sPHENIX layer
346  // r2, x2 and y2 are parameters of circle fitted to TPC clusters
347  // the solutions are xplus, xminus, yplus, yminus
348 
349  // The intersection of two circles occurs when
350  // (x-x1)^2 + (y-y1)^2 = r1^2, / (x-x2)^2 + (y-y2)^2 = r2^2
351  // Here we assume that circle 1 is an sPHENIX layer centered on x1=y1=0, and circle 2 is arbitrary
352  // x^2 +y^2 = r1^2, (x-x2)^2 + (y-y2)^2 = r2^2
353  // expand the equations and subtract to eliminate the x^2 and y^2 terms, gives the radical line connecting the intersection points
354  // iy = - (2*x2*x - D) / 2*y2,
355  // then substitute for y in equation of circle 1
356 
357  double D = r1*r1 - r2*r2 + x2*x2 + y2*y2;
358  double a = 1.0 + (x2*x2) / (y2*y2);
359  double b = - D * x2/( y2*y2);
360  double c = D*D / (4.0*y2*y2) - r1*r1;
361 
362  xplus = (-b + sqrt(b*b - 4.0* a * c) ) / (2.0 * a);
363  xminus = (-b - sqrt(b*b - 4.0* a * c) ) / (2.0 * a);
364 
365  // both values of x are valid
366  // but for each of those values, there are two possible y values on circle 1
367  // but only one of those falls on the radical line:
368 
369  yplus = - (2*x2*xplus - D) / (2.0*y2);
370  yminus = -(2*x2*xminus - D) / (2.0*y2);
371 
372 }

374 void PHTpcClusterMover::line_fit(std::vector<TrkrCluster*> clusters, double &a, double &b)
375 {
376  // copied from: https://www.bragitoff.com
377  // we want to fit z vs radius
378 
379  double xsum=0,x2sum=0,ysum=0,xysum=0; //variables for sums/sigma of xi,yi,xi^2,xiyi etc
380  for (unsigned int i=0; i<clusters.size(); ++i)
381  {
382  double z = clusters[i]->getZ();
383  double r = sqrt(pow(clusters[i]->getX(),2) + pow(clusters[i]->getY(), 2));
384 
385  xsum=xsum+r; //calculate sigma(xi)
386  ysum=ysum+z; //calculate sigma(yi)
387  x2sum=x2sum+pow(r,2); //calculate sigma(x^2i)
388  xysum=xysum+r*z; //calculate sigma(xi*yi)
389  }
390  a=(clusters.size()*xysum-xsum*ysum)/(clusters.size()*x2sum-xsum*xsum); //calculate slope
391  b=(x2sum*ysum-xsum*xysum)/(x2sum*clusters.size()-xsum*xsum); //calculate intercept
392 
393  if(Verbosity() > 10)
394  {
395  for (unsigned int i=0;i<clusters.size(); ++i)
396  {
397  double r = sqrt(pow(clusters[i]->getX(),2) + pow(clusters[i]->getY(), 2));
398  double z_fit = a * r + b; //to calculate y(fitted) at given x points
399  std::cout << " r " << r << " z " << clusters[i]->getZ() << " z_fit " << z_fit << std::endl;
400  }
401  }
402 
403  return;
