167 int MicromegasMapping::get_physical_strip( int fee_id, int channel_id) const
168 {
169  // bound check
170  if( channel_id < 0 || channel_id >= MicromegasDefs::m_nchannels_fee )
171  {
172  std::cout << "MicromegasMapping::get_physical_strip - invalid channel: " << channel_id << std::endl;
173  return -1;
174  }
175 
176  // get hitsetkey and orientation
177  const auto hitsetkey = get_hitsetkey(fee_id);
178  const auto segmentation_type = MicromegasDefs::getSegmentationType(hitsetkey);
179  switch (segmentation_type)
180  {
181  case MicromegasDefs::SegmentationType::SEGMENTATION_Z:
182  return m_fee_to_strip_mapping_z[channel_id];
183 
184  case MicromegasDefs::SegmentationType::SEGMENTATION_PHI:
185  return m_fee_to_strip_mapping_phi[channel_id];
186  }
187 
188  // never reached
189  return -1;
190 }
