556 {
557  double l1 = sqrt(x1*x1+y1*y1+z1*z1);
558  double l2 = sqrt(x2*x2+y2*y2+z2*z2);
559  double sx = (x1/l1+x2/l2);
560  double sy = (y1/l1+y2/l2);
561  double sz = (z1/l1+z2/l2);
562  double dx = (x1/l1-x2/l2);
563  double dy = (y1/l1-y2/l2);
564  double dz = (z1/l1-z2/l2);
565  return 2*atan2(sqrt(dx*dx+dy*dy+dz*dz),sqrt(sx*sx+sy*sy+sz*sz));
566 }

1004  layer-0.5,
1005  brokenCandidates);

1008  // find corresponding seed for starting cluster

1010  [&](const pointKey &k){return cand.second == k.second;});
1011  if(iter == frontEnds.end()) continue;

1016  mergedseed.insert(mergedseed.end(),seeds[seed_index].begin(),seeds[seed_index].end());
1017  merged.push_back(mergedseed);
1018  // mark seeds as used
1019  isUsed[i] = true;
1020  isUsed[seed_index] = true;
1021  }
1022  }
1023  // include unused seeds as-is
1024  for(size_t i=0;i<seeds.size();i++)
1025  {
1026  if(!isUsed[i]) merged.push_back(seeds[i]);
1027  }
1028  return merged;
1029 }
