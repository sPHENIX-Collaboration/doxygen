60 int main(int argc, char* argv[]) {
61  using namespace trento;
62 
63  // Parse options with boost::program_options.
64  // There are quite a few options, so let's separate them into logical groups.
65  using OptDesc = po::options_description;
66 
67  using VecStr = std::vector<std::string>;
68  OptDesc main_opts{};
69  main_opts.add_options()
70  ("projectile", po::value<VecStr>()->required()->
71  notifier( // use a lambda to verify there are exactly two projectiles
72  [](const VecStr& projectiles) {
73  if (projectiles.size() != 2)
74  throw po::required_option{"projectile"};
75  }),
76  "projectile symbols")
77  ("number-events", po::value<int>()->default_value(1),
78  "number of events");
79 
80  // Make all main arguments positional.
81  po::positional_options_description positional_opts{};
82  positional_opts
83  .add("projectile", 2)
84  .add("number-events", 1);
85 
86  using VecPath = std::vector<fs::path>;
87  OptDesc general_opts{"general options"};
88  general_opts.add_options()
89  ("help,h", "show this help message and exit")
90  ("version", "print version information and exit")
91  ("bibtex", "print bibtex entry and exit")
92  // ("default-config", "print a config file with default settings and exit")
93  ("config-file,c", po::value<VecPath>()->value_name("FILE"),
94  "configuration file\n(can be passed multiple times)");
95 
96  OptDesc output_opts{"output options"};
97  output_opts.add_options()
98  ("quiet,q", po::bool_switch(),
99  "do not print event properties to stdout")
100  ("output,o", po::value<fs::path>()->value_name("PATH"),
101  "HDF5 file or directory for text files");
102 
103  OptDesc phys_opts{"physical options"};
104  phys_opts.add_options()
105  ("reduced-thickness,p",
106  po::value<double>()->value_name("FLOAT")->default_value(0., "0"),
107  "reduced thickness parameter")
108  ("fluctuation,k",
109  po::value<double>()->value_name("FLOAT")->default_value(1., "1"),
110  "gamma fluctuation shape parameter")
111  ("nucleon-width,w",
112  po::value<double>()->value_name("FLOAT")->default_value(.5, "0.5"),
113  "Gaussian nucleon width [fm]")
114  ("cross-section,x",
115  po::value<double>()->value_name("FLOAT")->default_value(6.4, "6.4"),
116  "inelastic nucleon-nucleon cross section sigma_NN [fm^2]")
117  ("normalization,n",
118  po::value<double>()->value_name("FLOAT")->default_value(1., "1"),
119  "normalization factor")
120  ("b-min",
121  po::value<double>()->value_name("FLOAT")->default_value(0., "0"),
122  "minimum impact parameter [fm]")
123  ("b-max",
124  po::value<double>()->value_name("FLOAT")->default_value(-1., "auto"),
125  "maximum impact parameter [fm]")
126  ("random-seed",
127  po::value<int64_t>()->value_name("INT")->default_value(-1, "auto"),
128  "random seed");
129 
130  OptDesc grid_opts{"grid options"};
131  grid_opts.add_options()
132  ("grid-max",
133  po::value<double>()->value_name("FLOAT")->default_value(10., "10.0"),
134  "xy max [fm]\n(grid extends from -max to +max)")
135  ("grid-step",
136  po::value<double>()->value_name("FLOAT")->default_value(0.2, "0.2"),
137  "step size [fm]");
138 
139  // Make a meta-group containing all the option groups except the main
140  // positional options (don't want the auto-generated usage info for those).
141  OptDesc usage_opts{};
142  usage_opts
143  .add(general_opts)
144  .add(output_opts)
145  .add(phys_opts)
146  .add(grid_opts);
147 
148  // Now a meta-group containing _all_ options.
149  OptDesc all_opts{};
150  all_opts
151  .add(usage_opts)
152  .add(main_opts);
153 
154  // Will be used several times.
155  const std::string usage_str{
156  "usage: trento [options] projectile projectile [number-events = 1]\n"};
157 
158  try {
159  // Initialize a VarMap (boost::program_options::variables_map).
160  // It will contain all configuration values.
161  VarMap var_map{};
162 
163  // Parse command line options.
164  po::store(po::command_line_parser(argc, argv)
165  .options(all_opts).positional(positional_opts).run(), var_map);
166 
167  // Handle options that imply immediate exit.
168  // Must do this _before_ po::notify() since that can throw exceptions.
169  if (var_map.count("help")) {
170  std::cout
171  << usage_str
172  << "\n"
173  "projectile = { p | d | Cu | Cu2 | Au | Au2 | Pb | U | U2 | U3 }\n"
174  << usage_opts
175  << "\n"
176  "see the online documentation for complete usage information\n";
177  return 0;
178  }
179  if (var_map.count("version")) {
180  print_version();
181  return 0;
182  }
183  if (var_map.count("bibtex")) {
184  print_bibtex();
185  return 0;
186  }
187  // if (var_map.count("default-config")) {
188  // print_default_config();
189  // return 0;
190  // }
191 
192  // Merge any config files.
193  if (var_map.count("config-file")) {
194  // Everything except general_opts.
195  OptDesc config_file_opts{};
196  config_file_opts
197  .add(main_opts)
198  .add(output_opts)
199  .add(phys_opts)
200  .add(grid_opts);
201 
202  for (const auto& path : var_map["config-file"].as<VecPath>()) {
203  if (!fs::exists(path)) {
204  throw po::error{
205  "configuration file '" + path.string() + "' not found"};
206  }
207  fs::ifstream ifs{path};
208  po::store(po::parse_config_file(ifs, config_file_opts), var_map);
209  }
210  }
211 
212  // Save all the final values into var_map.
213  // Exceptions may occur here.
214  po::notify(var_map);
215 
216  // Go!
217  Collider collider{var_map};
218  collider.run_events();
219  } catch (const po::required_option&) {
220  // Handle this exception separately from others.
221  // This occurs e.g. when the program is excuted with no arguments.
222  std::cerr << usage_str << "run 'trento --help' for more information\n";
223  return 1;
224  } catch (const std::exception& e) {
225  // For all other exceptions just output the error message.
226  std::cerr << e.what() << '\n';
227  return 1;
228  }
229 
230  return 0;
231 }
