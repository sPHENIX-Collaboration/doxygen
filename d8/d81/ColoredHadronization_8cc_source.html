98 void ColoredHadronization::DoHadronization(vector<vector<shared_ptr<Parton>>>& shower, vector<shared_ptr<Hadron>>& hOut, vector<shared_ptr<Parton>>& pOut)
99 {
100 
101  Event& event = pythia.event;
102  event.reset();
103  double pz = p_fake;
104 
105  JSDEBUG << "&&&&&&&&&&&&&&&&&&& the number of showers are: " << shower.size();
106  for(unsigned int ishower=0; ishower < shower.size(); ++ishower)
107  {
108  JSDEBUG << "&&&&&&&&&&&&&&&&&&& there are " << shower.at(ishower).size() << " partons in the shower number " << ishower;
109  for(unsigned int ipart=0; ipart < shower.at(ishower).size(); ++ipart)
110  {
111  double onshellE = pow(pow(shower.at(ishower).at(ipart)->px(),2) + pow(shower.at(ishower).at(ipart)->py(),2) + pow(shower.at(ishower).at(ipart)->pz(),2) ,0.5 ) ;
112  event.append(shower.at(ishower).at(ipart)->pid(),23,shower.at(ishower).at(ipart)->color(),shower.at(ishower).at(ipart)->anti_color(),
113  shower.at(ishower).at(ipart)->px(),shower.at(ishower).at(ipart)->py(),shower.at(ishower).at(ipart)->pz(),onshellE);
114  }
115  unsigned int color, anti_color;
116  int pid;
117 
118  anti_color = shower.at(ishower).at(0)->min_anti_color();
119  color = shower.at(ishower).at(0)->min_color();
120 
121  if ((color>100)&&(anti_color>100))
122  {
123  pid = 21;
124  }
125  else if ((color>100)&&(anti_color<100))
126  {
127  pid = -1;
128  }
129  else
130  {
131  pid = 1;
132  }
133 
134  pz = -1*pz;
135  event.append(pid, 23, anti_color, color, 0.2, 0.2, pz, sqrt(pz*pz + 0.08));
136 
137  VERBOSE(2) <<"There are " << hOut.size() << " Hadrons and " << pOut.size() << " partons after Hadronization";
138  }
139 
140  pythia.next();
141  // event.list();
142 
143  unsigned int ip=hOut.size();
144  for (unsigned int i=0; i<event.size(); ++i){
145  if ( !event[i].isFinal() ) continue;
146  if ( !event[i].isHadron() ) continue;
147  if(fabs(event[i].eta())>20) continue; //To prevent "nan" from propagating, very rare though
148 
149  double x[4] = {0,0,0,0};
150  hOut.push_back(make_shared<Hadron>(ip,event[i].id(),event[i].status(),event[i].pT(),event[i].eta(), event[i].phi(), event[i].e(), x));
151  ++ip;
152  }
153 
154  shower.clear();
155 
156 }
