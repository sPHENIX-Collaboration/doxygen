170  EventNumber++;
171 
172  cout<<"--------------------------- EventNumber = " << EventNumber-1 << endl;
173  if(EventNumber==1) topNode->print();
174 
175  PHHepMCGenEventMap *genevtmap = findNode::getClass<PHHepMCGenEventMap>(topNode, "PHHepMCGenEventMap");
176  PHHepMCGenEvent *genevt = nullptr;
177 
178  for (PHHepMCGenEventMap::ReverseIter iter = genevtmap->rbegin(); iter != genevtmap->rend(); ++iter)
179  {
180  genevt = iter->second;
181  if(!genevt) {cout<<"no phgenevt!" << endl; return Fun4AllReturnCodes::ABORTEVENT;}
182  cout << "got PHHepMCGenEvent... " << genevt << endl;
183  }
184 
185  HepMC::GenEvent *event = genevt->getEvent();
186  if (!event) { cout << PHWHERE << "no genevent!" << endl; return Fun4AllReturnCodes::ABORTEVENT;}
187  int npart = event->particles_size();
188  int nvert = event->vertices_size();
189  cout << "Number of particles = " << npart << " " << nvert << endl;

191 /*
192  for (HepMC::GenEvent::particle_const_iterator p = event->particles_begin(); p != event->particles_end(); ++p)
193  {
194  int pid = (*p)->pdg_id();
195  int status = (*p)->status();
196  double pt = ((*p)->momentum()).perp();
197 // double mass = ((*p)->momentum()).m();

199  if(pt>2.0) cout << pid << " " << pt << " " << status << endl;
200  }
201 */
202 
203 // generate Upsilon
204 
205  double pt = _rng->Uniform() * 10.;
206  double y = (_rng->Uniform()*2.) - 1.;
207  double phi = (2.0 * M_PI) * _rng->Uniform();
208  double mass = 9.3987;
209  double mt = sqrt(mass * mass + pt * pt);
210  double eta = asinh(sinh(y) * mt / pt);
211  TLorentzVector vm;
212  vm.SetPtEtaPhiM(pt, eta, phi, mass);
213 
214 // decay it to electrons
215 
216  double m1 = 0.000511;
217  double m2 = 0.000511;
218 
219 // first electroin in Upsilon rest frame
220  double E1 = (mass * mass - m2 * m2 + m1 * m1) / (2.0 * mass);
221  double p1 = sqrt((mass * mass - (m1 + m2) * (m1 + m2)) * (mass * mass - (m1 - m2) * (m1 - m2))) / (2.0 * mass);
222  double phi1 = _rng->Uniform() * 2. * M_PI;
223  double th1 = fsin->GetRandom();
224  double px1 = p1 * sin(th1) * cos(phi1);
225  double py1 = p1 * sin(th1) * sin(phi1);
226  double pz1 = p1 * cos(th1);
227  TLorentzVector v1;
228  v1.SetPxPyPzE(px1, py1, pz1, E1);
229 
230 // boost to lab frame
231  double betax = vm.Px() / vm.E();
232  double betay = vm.Py() / vm.E();
233  double betaz = vm.Pz() / vm.E();
234  v1.Boost(betax, betay, betaz);
235 
236 // second electron
237  TLorentzVector v2 = vm - v1;
238  cout << "decay electrons: " << v1.Pt() << " " << v2.Pt() << endl;
239 
240  HepMC::FourVector fv1 = HepMC::FourVector(v1.Px(),v1.Py(),v1.Pz());
241  HepMC::GenParticle gp1 = HepMC::GenParticle(fv1,-11);
242  HepMC::FourVector fv2 = HepMC::FourVector(v2.Px(),v2.Py(),v2.Pz());
243  HepMC::GenParticle gp2 = HepMC::GenParticle(fv2,11);
244 
245  HepMC::GenEvent* newevent = new HepMC::GenEvent(*event);
246 
247  for (HepMC::GenEvent::vertex_iterator v = newevent->vertices_begin(); v != newevent->vertices_end(); ++v)
248  {
249  int npin = (*v)->particles_in_size();
250  int npout = (*v)->particles_out_size();
251  HepMC::ThreeVector pnt = (*v)->point3d();
252  cout << "event vertex: " << npin << " " << npout << " " << pnt.x() << " " << pnt.y() << " " << pnt.z() << endl;
253  cout << "adding particle with id " << gp1.pdg_id() << endl;
254  (*v)->add_particle_out(&gp1);
255  cout << " after adding electron1: " << (*v)->particles_out_size() << endl;
256  (*v)->add_particle_out(&gp2);
257  cout << " after adding electron2: " << (*v)->particles_out_size() << endl;
258  break; // add electrons to the first vertex
259  }
260 
261  PHHepMCGenEvent* newgenevt = new PHHepMCGenEvent();
262  newgenevt->addEvent(newevent);
263 
264 // genevt->clearEvent();
265  PHHepMCGenEvent* tmpptr = genevtmap->insert_event(1, newgenevt);
266  cout << "inserted event " << tmpptr << endl;
267 
268  cout << "procedd_event_hepmc() ended." << endl;
269  return Fun4AllReturnCodes::EVENT_OK;
270 }

287 
288  for (sPHElectronPairContainerv1::ConstIter iter = eePairs->begin(); iter != eePairs->end(); ++iter)
289  {
290  sPHElectronPairv1* pair = iter->second;
291  sPHElectronv1* electron = (sPHElectronv1*)pair->get_first();
292  sPHElectronv1* positron = (sPHElectronv1*)pair->get_second();
293  tmp[0] = pair->get_type();
294  tmp[1] = pair->get_mass();
295  tmp[2] = pair->get_pt();
296  tmp[3] = pair->get_eta();
297  double px1 = electron->get_px();
298  double py1 = electron->get_py();
299  double pz1 = electron->get_pz();
300  double px2 = positron->get_px();
301  double py2 = positron->get_py();
302  double pz2 = positron->get_pz();
303  tmp[4] = sqrt(px1*px1+py1*py1);
304  tmp[5] = sqrt(px2*px2+py2*py2);
305  tmp[6] = electron->get_e3x3();
306  tmp[7] = positron->get_e3x3();
307  tmp[8] = electron->get_emce();
308  tmp[9] = positron->get_emce();
309  tmp[10] = sqrt(px1*px1+py1*py1+pz1*pz1);
310  tmp[11] = sqrt(px2*px2+py2*py2+pz2*pz2);
311  double chisq1 = electron->get_chi2();
312  double ndf1 = (double)electron->get_ndf();
313  if(ndf1!=0.) { chisq1 = chisq1/ndf1;} else {chisq1 = 99999.;}
314  double chisq2 = positron->get_chi2();
315  double ndf2 = (double)positron->get_ndf();
316  if(ndf2!=0.) { chisq2 = chisq2/ndf2;} else {chisq2 = 99999.;}
317  tmp[12] = chisq1;
318  tmp[13] = chisq2;
319  tmp[14] = electron->get_dca2d();
320  tmp[15] = positron->get_dca2d();
321  tmp[16] = electron->get_dca3d_xy();
322  tmp[17] = positron->get_dca3d_xy();
323  tmp[18] = electron->get_dca3d_z();
324  tmp[19] = positron->get_dca3d_z();
325  tmp[20] = mult;
326  tmp[21] = 0.; // rapidity
327  tmp[22] = positron->get_nmvtx();
328  tmp[23] = electron->get_nmvtx();
329  tmp[24] = positron->get_ntpc();
330  tmp[25] = electron->get_ntpc();
331 
332  ntp1->Fill(tmp);
333  }
334 
335 
336  cout << "process_event_pairs() ended." << endl;
337  return Fun4AllReturnCodes::EVENT_OK;
338 }

681 bool sPHAnalysis::isElectron(SvtxTrack* trk)
682 {
683  double px = trk->get_px();
684  double py = trk->get_py();
685  double pz = trk->get_pz();
686  double pt = sqrt(px*px+py*py);
687  double pp = sqrt(pt*pt+pz*pz);
688  double e3x3 = trk->get_cal_energy_3x3(SvtxTrack::CAL_LAYER::CEMC);
689  //double eclus = trk->get_cal_cluster_e(SvtxTrack::CAL_LAYER::CEMC);
690  if(pp==0.) return false;
691  //if(pt<2.0) return false;
692  if(pt<0.1) return false;
693  //cout << e3x3 << " " << eclus << endl;
694  if(isnan(e3x3)) return false;
695  //if(e3x3/pp<0.70) return false;
696  if(e3x3/pp<0.1) return false;
697  return true;
698 }
