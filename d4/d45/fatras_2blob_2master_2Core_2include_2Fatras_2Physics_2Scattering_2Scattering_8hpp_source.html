46  // Do nothing if the flag is set to false

54  // parametric scattering

56 
57  // the initial values
58  double theta = Acts::VectorHelpers::theta(in.momentum());
59  double phi = Acts::VectorHelpers::phi(in.momentum());
60  double sinTheta = (sin(theta) * sin(theta) > 10e-10) ? sin(theta) : 1.;
61 
62  // sample them in an independent way
63  double deltaTheta = projectionFactor * angle3D;
64  double numDetlaPhi = 0.; //?? @THIS IS WRONG HERE !
65  double deltaPhi = projectionFactor * numDetlaPhi / sinTheta;
66 
67  // @todo: use bound parameter
68  // (i) phi
69  phi += deltaPhi;
70  if (phi >= M_PI)
71  phi -= M_PI;
72  else if (phi < -M_PI)
73  phi += M_PI;
74  // (ii) theta
75  theta += deltaTheta;
76  if (theta > M_PI)
77  theta -= M_PI;
78  else if (theta < 0.)
79  theta += M_PI;
80 
81  double sphi = std::sin(phi);
82  double cphi = std::cos(phi);
83  double stheta = std::sin(theta);
84  double ctheta = std::cos(theta);
85 
86  // assign the new values
87  in.scatter(in.p() * Acts::Vector3D(cphi * stheta, sphi * stheta, ctheta));
88  } else {

94  double psi = 2. * M_PI * uniformDist(gen);
95 
96  // more complex but "more true"
97  Acts::Vector3D pDirection(in.momentum().normalized());

99  double y = pDirection.x();

101

112  Acts::AngleAxis3D(angle3D, deflector);

115  }
