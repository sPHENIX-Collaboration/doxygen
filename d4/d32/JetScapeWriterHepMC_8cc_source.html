31  void JetScapeWriterHepMC::WriteHeaderToFile() {
32  // Create event here - not actually writing
33  // TODO: GeV seems right, but I don't think we actually measure in mm
34  // Should multiply all lengths by 1e-12 probably
35  evt = GenEvent(Units::GEV,Units::MM);
36 
37  // Expects pb, pythia delivers mb
38  auto xsec = make_shared<HepMC::GenCrossSection>();
39  xsec->set_cross_section( GetHeader().GetSigmaGen() * 1e9, 0);
40  xsec->set_cross_section( GetHeader().GetSigmaGen() * 1e9, GetHeader().GetSigmaErr() * 1e9);
41  evt.set_cross_section( xsec );
42  evt.weights().push_back( GetHeader().GetEventWeight() );
43 
44  auto heavyion = make_shared<HepMC::GenHeavyIon>();
45  // see https://gitlab.cern.ch/hepmc/HepMC3/blob/master/include/HepMC/GenHeavyIon.h
46  if ( GetHeader().GetNpart() > -1 ){
47  // Not clear what the difference is...
48  heavyion->Ncoll_hard = GetHeader().GetNcoll();
49  heavyion->Ncoll = GetHeader().GetNcoll();
50  }
51  if ( GetHeader().GetNcoll() > -1 ){
52  // Hepmc separates into target and projectile.
53  // Set one? Which? Both? half to each? setting projectile for now.
54  // setting both might lead to weird problems when they get added up
55  heavyion->Npart_proj = GetHeader().GetNpart();
56  }
57  if ( GetHeader().GetTotalEntropy() > -1 ){
58  // nothing good in the HepMC standard. Something related to mulitplicity would work
59  }
60 
61  if ( GetHeader().GetEventPlaneAngle() > -999 ){
62  heavyion->event_plane_angle = GetHeader().GetEventPlaneAngle();
63  }
64 
65  evt.set_heavy_ion( heavyion );
66  }

81  void JetScapeWriterHepMC::Write(weak_ptr<PartonShower> ps){
82  shared_ptr<PartonShower> pShower = ps.lock();
83  if ( !pShower ) return;
84 
85  // Need topological order, see
86  // https://hepmc.web.cern.ch/hepmc/differences.html
87  // That means if parton p1 comes into vertex v, and p2 goes out of v,
88  // then p1 has to be created (bestowed an id) before p2
89  // Take inspiration from hepmc3.0.0/interfaces/pythia8/src/Pythia8ToHepMC3.cc
90  // But pythia showers are different from our existing graph structure,
91  // So instead try to modify the first attempt to respect top. order
92  // and don't create vertices and particles more than once
93 
94  // Using GTL's topsort
95  // 1. Check that our graph is sane
96  if ( !pShower->is_acyclic() ) throw std::runtime_error("PROBLEM in JetScapeWriterHepMC: Graph is not acyclic.");
97 
98  // 2.
99  topsort topsortsearch;
100  topsortsearch.scan_whole_graph(true);
101  topsortsearch.start_node(); // defaults to first node
102  topsortsearch.run(*pShower);
103  auto nEnd = topsortsearch.top_order_end(); // this is a topsort::topsort_iterator
104 
105  // Need to keep track of already created ones
106  map< int , GenParticlePtr > CreatedPartons;
107 
108  // probably unnecessary, used for consistency checks
109  map< int , bool > vused;
110  bool foundRoot=false;
111  for ( auto nIt = topsortsearch.top_order_begin(); nIt != nEnd; ++nIt){
112  // cout << *nIt << " " << nIt->indeg() << " " << nIt->outdeg() << endl;
113 
114  // Should be the only time we see this node.
115  if ( vused.find ( nIt->id() ) != vused.end() ) throw std::runtime_error("PROBLEM in JetScapeWriterHepMC: Reusing a vertex.");
116  vused[ nIt->id() ]=true;
117 
118  // 1. Create a new vertex.
119  // --------------------------------------------
120  auto v = castVtxToHepMC( pShower->GetVertex( *nIt ) );
121 
122  // 2. Incoming edges?
123  // --------------------------------------------
124  // 2.1: No. Need to create one (thanks, HepMC)
125  if ( nIt->indeg() == 0 ){
126  if (foundRoot){ // Should only happen once unless we merged showers
127  WARN << "Found a second root! Should only happen if we merged showers. "
128  << " If that's the case, please comment out the following throw and recompile";
129  throw std::runtime_error("PROBLEM in JetScapeWriterHepMC: Found a second root.");
130  }
131  foundRoot=true;
132 
133  // Some choices here. In general, could just attach to a dummy.
134  // Maybe better: The root should only have exactly one daughter, let's check and clone
135  if ( nIt->outdeg() != 1 ){
136  throw std::runtime_error("PROBLEM in JetScapeWriterHepMC: Root has illegal number of daughters");
137  }
138  auto out = pShower->GetParton( *(nIt->out_edges_begin()) );
139  auto hepin = castPartonToHepMC( out );
140  v->add_particle_in( hepin );
141  }
142 
143  // 2.2: Have incoming edges.
144  if ( nIt->indeg() > 0 ){
145  // In the current framework, it should only be one.
146  // So we will catch anything more but provide a mechanism that should work anyway.
147  if (nIt->indeg() > 1 ){
148  WARN << "Found more than one mother parton! Should only happen if we added medium particles. "
149  << "The code should work, but proceed with caution";
150  }
151 
152  auto inIt = nIt->in_edges_begin();
153  auto inEnd = nIt->in_edges_end();
154  for ( /* nop */; inIt != inEnd; ++inIt ){
155  auto phepin = CreatedPartons.find ( inIt->id() );
156  if ( phepin != CreatedPartons.end() ){
157  // We should already have one!
158  v->add_particle_in( phepin->second );
159  } else {
160  WARN << "Incoming particle out of nowhere. This could maybe happen if we pick up medium particles "
161  << " but is probably a topsort problem. Try using the code after this throw() but be very careful.";
162  throw std::runtime_error("PROBLEM in JetScapeWriterHepMC: Incoming particle out of nowhere.");
163 
164  auto in = pShower->GetParton( *inIt );
165  auto hepin = castPartonToHepMC( in );
166  CreatedPartons [ inIt->id() ] = hepin;
167  v->add_particle_in( hepin );
168  }
169  }
170  }
171 
172  // 3. Outgoing edges?
173  // --------------------------------------------
174  // 3.1: No. Need to create one (thanks, HepMC)
175  if ( nIt->outdeg() == 0 ){
176  // Some choices here. In general, could just attach to a dummy.
177  // That may be the best option, but it is hard to imagine a physics scenario
178  // where a final vertex should have more than one parent
179  // So let's clone the incoming guy
180  if ( nIt->indeg() != 1 ){
181  throw std::runtime_error("PROBLEM in JetScapeWriterHepMC: Need exactly one parent to clone final state partons.");
182  }
183  auto in = pShower->GetParton( *(nIt->in_edges_begin()) );
184  auto hepout = castPartonToHepMC( in );
185  v->add_particle_out( hepout );
186  // Note that we do not register this particle. Since it's pointing nowhere it can never be reused.
187  }
188 
189  // 3.2: Otherwise use it and register it
190  if ( nIt->outdeg() > 0 ){
191  auto outIt = nIt->out_edges_begin();
192  auto outEnd = nIt->out_edges_end();
193  for ( /* nop */; outIt != outEnd; ++outIt ){
194  if ( CreatedPartons.find ( outIt->id() ) != CreatedPartons.end() ){
195  throw std::runtime_error("PROBLEM in JetScapeWriterHepMC: Trying to recreate a preexisting GenParticle.");
196  }
197  auto out = pShower->GetParton( *outIt );
198  auto hepout = castPartonToHepMC( out );
199  CreatedPartons [ outIt->id() ] = hepout;
200  v->add_particle_out( hepout );
201  }
202  }
203 
204  vertices.push_back ( v );
205  }
206  }

208  void JetScapeWriterHepMC::Write(weak_ptr<Hadron> h)
209  {
210  auto hadron = h.lock();
211  if ( !hadron ) return;
212 
213  // No clear source for most hadrons
214  // Also, a graph with e.g. recombination hadrons would have loops,
215  // (though the direction should still make it acyclic?)
216  // Not sure how this is supposed to be done in HepMC3
217  // Our solution: Attach all hadrons to one dedicated hadronization vertex.
218  // Future option: Have separate shower and bulk vertices?
219 
220  // Create if it doesn't exist yet
221  if ( !hadronizationvertex ) {
222  // dummy position
223  HepMC::FourVector vtxPosition( 0,0,0, 100 ); // set it to a late time...
224  hadronizationvertex = make_shared<GenVertex>(vtxPosition);
225 
226  // dummy mother -- could also maybe use the first/hardest shower initiator
227  HepMC::FourVector pmom(0, 0, 0, 0);
228  make_shared<GenParticle> (pmom, 0, 0);
229  hadronizationvertex->add_particle_in( make_shared<GenParticle> (pmom, 0, 0) );
230 
231  vertices.push_back ( hadronizationvertex );
232  }
233 
234  // now attach
235  hadronizationvertex->add_particle_out( castHadronToHepMC( hadron ) );
236  }
