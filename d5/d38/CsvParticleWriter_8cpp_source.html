34 ActsExamples::ProcessCode ActsExamples::CsvParticleWriter::writeT(
35  const ActsExamples::AlgorithmContext& ctx,
36  const SimParticleContainer& particles) {
37  auto pathParticles = perEventFilepath(
38  m_cfg.outputDir, m_cfg.outputStem + ".csv", ctx.eventNumber);
39  dfe::NamedTupleCsvWriter<ParticleData> writer(pathParticles,
40  m_cfg.outputPrecision);
41 
42  ParticleData data;
43  for (const auto& particle : particles) {
44  data.particle_id = particle.particleId().value();
45  data.particle_type = particle.pdg();
46  data.process = static_cast<decltype(data.process)>(particle.process());
47  data.vx = particle.position().x() / Acts::UnitConstants::mm;
48  data.vy = particle.position().y() / Acts::UnitConstants::mm;
49  data.vz = particle.position().z() / Acts::UnitConstants::mm;
50  data.vt = particle.time() / Acts::UnitConstants::ns;
51  const auto p = particle.absoluteMomentum() / Acts::UnitConstants::GeV;
52  data.px = p * particle.direction().x();
53  data.py = p * particle.direction().y();
54  data.pz = p * particle.direction().z();
55  data.m = particle.mass() / Acts::UnitConstants::GeV;
56  data.q = particle.charge() / Acts::UnitConstants::e;
57  writer.append(data);
58  }
59 
60  return ProcessCode::SUCCESS;
61 }
