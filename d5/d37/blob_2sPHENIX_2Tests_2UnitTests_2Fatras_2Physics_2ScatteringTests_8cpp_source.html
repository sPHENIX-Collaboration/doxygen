26 // Common test method that will be instantiated for each scattering model.
27 template <typename Scattering>
28 void test(const Scattering& scattering, uint32_t seed,
29  const ActsFatras::Particle& before) {
30  std::ranlux48 gen(seed);
31  ActsFatras::Particle after = before;
32 
33  const auto outgoing = scattering(gen, Acts::Test::makePercentSlab(), after);
34  // scattering leaves absolute energy/momentum unchanged
35  CHECK_CLOSE_REL(after.absoluteMomentum(), before.absoluteMomentum(), eps);
36  CHECK_CLOSE_REL(after.energy(), before.energy(), eps);
37  // scattering has changed the direction
38  BOOST_CHECK_LT(before.direction().dot(after.direction()), 1);
39  // scattering creates no new particles
40  BOOST_CHECK(outgoing.empty());
41 }
42 } // namespace
