73  inline FluidCellInfo operator+(FluidCellInfo a, const FluidCellInfo & b) {
74  a.energy_density += b.energy_density;

238  void CheckInRange(Jetscape::real tau, Jetscape::real x, Jetscape::real y, Jetscape::real eta) {
239  if (tau < tau_min || tau > TauMax()) {
240  throw InvalidSpaceTimeRange("tau=" + std::to_string(tau)
241  + " is not in range [" + std::to_string(tau_min) + ","
242  + std::to_string(TauMax()) + "]");
243  }
244  if (x < x_min || x > XMax()) {
245  throw InvalidSpaceTimeRange("x=" + std::to_string(x)
246  + " is not in range [" + std::to_string(x_min) + ","
247  + std::to_string(XMax()) + "]");
248  }
249  if (y < y_min || y > YMax()) {
250  throw InvalidSpaceTimeRange("y=" + std::to_string(y)
251  + " is not in range [" + std::to_string(y_min) + ","
252  + std::to_string(YMax()) + "]");
253  }
254  if (eta < eta_min || eta > EtaMax()) {
255  throw InvalidSpaceTimeRange("eta=" + std::to_string(eta)
256  + " is not in range [" + std::to_string(eta_min) + ","
257  + std::to_string(EtaMax()) + "]");
258  }
259  }
