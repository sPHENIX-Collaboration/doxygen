34 // Determine the maximum impact parameter. If the configuration contains a
35 // non-negative value for bmax, use it; otherwise, fall back to the minimum-bias
36 // default.
37 double determine_bmax(const VarMap& var_map,
38  const Nucleus& A, const Nucleus& B, const NucleonProfile& profile) {
39  auto bmax = var_map["b-max"].as<double>();
40  if (bmax < 0.)
41  bmax = A.radius() + B.radius() + profile.max_impact();
42  return bmax;
43 }

45 // Determine the asymmetry parameter (Collider::asymmetry_) for a pair of
46 // nuclei. It's just rA/(rA+rB), falling back to 1/2 if both radii are zero
47 // (i.e. for proton-proton).
48 double determine_asym(const Nucleus& A, const Nucleus& B) {
49  double rA = A.radius();
50  double rB = B.radius();
51  double sum = rA + rB;
52  if (sum < 0.1)
53  return 0.5;
54  else
55  return rA/sum;
56 }

111 void TrentoCollision::Sample(double smin, double smax){
112  double b;
113  while (true) {
114  b = SampleImpactParam();
115  // Pass the prepared nuclei to the Event. It computes the entropy profile
116  // (thickness grid) and other event observables.
117  event_.compute(*nucleusA_, *nucleusB_, nucleon_profile_);
118 
119  double total_entropy = event_.multiplicity();
120  if (total_entropy >= smin && total_entropy <= smax) break;
121  }
122 
123  info_.impact_parameter = b;
124  info_.num_participant = event_.npart();
125  info_.total_entropy = event_.multiplicity();
126  info_.ecc = event_.eccentricity();
127  info_.psi = event_.participant_plane();

129  // Write the mass center (x, y)

131  info_.ymid = event_.mass_center_index().second;

136 double TrentoCollision::SampleImpactParam() {
137  // Sample impact parameters until at least one nucleon-nucleon pair
138  // participates. The bool 'collision' keeps track -- it is effectively a
139  // logical OR over all possible participant pairs.
140  double b;
141  bool collision = false;
142 
143  do {
144  // Sample b from P(b)db = 2*pi*b.
145  b = bmin_ + (bmax_ - bmin_) * std::sqrt(random::canonical<double>());
146 
147  // Offset each nucleus depending on the asymmetry parameter (see header).
148  nucleusA_->sample_nucleons(asymmetry_ * b);
149  nucleusB_->sample_nucleons((asymmetry_ - 1.) * b);
150 
151  // Check each nucleon-nucleon pair.
152  for (auto&& A : *nucleusA_) {
153  for (auto&& B : *nucleusB_) {
154  collision = nucleon_profile_.participate(A, B) || collision;
155  }
156  }
157  } while (!collision);
158 
159  return b;
160 }
