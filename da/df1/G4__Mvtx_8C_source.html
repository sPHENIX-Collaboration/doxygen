67 double calculateArea(double inner_radius, double outer_radius) //Calculate the area of a disk

72 double calculateOR(double inner_radius, double area) //Calculate the outer radius of a disk, knowing the inner radius and the area

185 double Mvtx(PHG4Reco* g4Reco, double radius,
186  const int absorberactive = 0)
187 {
188  bool maps_overlapcheck = Enable::OVERLAPCHECK || Enable::MVTX_OVERLAPCHECK;
189  int verbosity = std::max(Enable::VERBOSITY, Enable::MVTX_VERBOSITY);
190 
191  PHG4MvtxSubsystem* mvtx = new PHG4MvtxSubsystem("MVTX");
192  mvtx->Verbosity(verbosity);
193 
194  for (int ilayer = 0; ilayer < G4MVTX::n_maps_layer; ilayer++)
195  {
196  double radius_lyr = PHG4MvtxDefs::mvtxdat[ilayer][PHG4MvtxDefs::kRmd];
197  if (verbosity)
198  {
199  cout << "Create Maps layer " << ilayer << " with radius " << radius_lyr << " mm." << endl;
200  }
201  radius = radius_lyr / 10.;
202  }
203  mvtx->set_string_param(PHG4MvtxDefs::GLOBAL, "stave_geometry_file", string(getenv("CALIBRATIONROOT")) + string("/Tracking/geometry/mvtx_stave_v1.gdml"));
204  mvtx->SetActive();
205  mvtx->OverlapCheck(maps_overlapcheck);
206  g4Reco->registerSubsystem(mvtx);
207  radius += G4MVTX::radius_offset;
208  if (Enable::MVTX_SERVICE) radius += MVTXService(g4Reco, radius);
209  return radius;
210 }
