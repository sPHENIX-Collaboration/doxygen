476 void PHTpcTrackSeedVertexAssoc::line_fit(std::vector<std::pair<double,double>> points, double &a, double &b)
477 {
478  // copied from: https://www.bragitoff.com
479  // we want to fit z vs radius
480 
481  double xsum=0,x2sum=0,ysum=0,xysum=0; //variables for sums/sigma of xi,yi,xi^2,xiyi etc
482  for (unsigned int i=0; i<points.size(); ++i)
483  {
484  double r = points[i].first;
485  double z = points[i].second;
486 
487  xsum=xsum+r; //calculate sigma(xi)
488  ysum=ysum+z; //calculate sigma(yi)
489  x2sum=x2sum+pow(r,2); //calculate sigma(x^2i)
490  xysum=xysum+r*z; //calculate sigma(xi*yi)
491  }
492  a=(points.size()*xysum-xsum*ysum)/(points.size()*x2sum-xsum*xsum); //calculate slope
493  b=(x2sum*ysum-xsum*xysum)/(x2sum*points.size()-xsum*xsum); //calculate intercept
494 
495  if(Verbosity() > 10)
496  {
497  for (unsigned int i=0;i<points.size(); ++i)
498  {
499  double r = points[i].first;
500  double z_fit = a * r + b; //to calculate z(fitted) at given r points
501  std::cout << " r " << r << " z " << points[i].second << " z_fit " << z_fit << std::endl;
502  }
503  }
504 
505  return;
506 }

526 /*
527  Circle fit to a given set of data points (in 2D)
528  This is an algebraic fit, due to Taubin, based on the journal article
529  G. Taubin, "Estimation Of Planar Curves, Surfaces And Nonplanar
530  Space Curves Defined By Implicit Equations, With
531  Applications To Edge And Range Image Segmentation",
532  IEEE Trans. PAMI, Vol. 13, pages 1115-1138, (1991)
533 */
534 {
535  int iter,IterMAX=99;
536 
537  double Mz,Mxy,Mxx,Myy,Mxz,Myz,Mzz,Cov_xy,Var_z;
538  double A0,A1,A2,A22,A3,A33;
539  double x,y;
540  double DET,Xcenter,Ycenter;
541 
542  // Compute x- and y- sample means
543  double meanX = 0;
544  double meanY = 0;
545  double weight = 0;
546  for(unsigned int i = 0; i < points.size(); ++i)
547  {
548  meanX += points[i].first;
549  meanY += points[i].second;
550  weight++;
551  }
552  meanX /= weight;
553  meanY /= weight;
554 
555  // computing moments
556 
557  Mxx=Myy=Mxy=Mxz=Myz=Mzz=0.;
558 
559  for (unsigned int i=0; i<points.size(); i++)
560  {
561  double Xi = points[i].first - meanX; // centered x-coordinates
562  double Yi = points[i].second - meanY; // centered y-coordinates
563  double Zi = Xi*Xi + Yi*Yi;
564 
565  Mxy += Xi*Yi;
566  Mxx += Xi*Xi;
567  Myy += Yi*Yi;
568  Mxz += Xi*Zi;
569  Myz += Yi*Zi;
570  Mzz += Zi*Zi;
571  }
572  Mxx /= weight;
573  Myy /= weight;
574  Mxy /= weight;
575  Mxz /= weight;
576  Myz /= weight;
577  Mzz /= weight;
578 
579  // computing coefficients of the characteristic polynomial
580 
581  Mz = Mxx + Myy;
582  Cov_xy = Mxx*Myy - Mxy*Mxy;
583  Var_z = Mzz - Mz*Mz;
584  A3 = 4*Mz;
585  A2 = -3*Mz*Mz - Mzz;
586  A1 = Var_z*Mz + 4*Cov_xy*Mz - Mxz*Mxz - Myz*Myz;
587  A0 = Mxz*(Mxz*Myy - Myz*Mxy) + Myz*(Myz*Mxx - Mxz*Mxy) - Var_z*Cov_xy;
588  A22 = A2 + A2;
589  A33 = A3 + A3 + A3;
590 
591  // finding the root of the characteristic polynomial
592  // using Newton's method starting at x=0
593  // (it is guaranteed to converge to the right root)
594 
595  for (x=0.,y=A0,iter=0; iter<IterMAX; iter++) // usually, 4-6 iterations are enough
596  {
597  double Dy = A1 + x*(A22 + A33*x);
598  double xnew = x - y/Dy;
599  if ((xnew == x)||(!isfinite(xnew))) break;
600  double ynew = A0 + xnew*(A1 + xnew*(A2 + xnew*A3));
601  if (fabs(ynew)>=fabs(y)) break;
602  x = xnew; y = ynew;
603  }
604 
605  // computing parameters of the fitting circle
606 
607  DET = x*x - x*Mz + Cov_xy;
608  Xcenter = (Mxz*(Myy - x) - Myz*Mxy)/DET/2;
609  Ycenter = (Myz*(Mxx - x) - Mxz*Mxy)/DET/2;
610 
611  // assembling the output
612 
613  X0 = Xcenter + meanX;
614  Y0 = Ycenter + meanY;
615  R = sqrt(Xcenter*Xcenter + Ycenter*Ycenter + Mz);
616 }

618 std::vector<double> PHTpcTrackSeedVertexAssoc::GetCircleClusterResiduals(std::vector<std::pair<double,double>> points, double R, double X0, double Y0)
619 {
620  std::vector<double> residues;
621  // calculate cluster residuals from the fitted circle
622  for(unsigned int i = 0; i < points.size(); ++i)
623  {
624  double x = points[i].first;
625  double y = points[i].second;
626 
627  // The shortest distance of a point from a circle is along the radial; line from the circle center to the point
628  double dca = sqrt( (x - X0)*(x-X0) + (y-Y0)*(y-Y0) ) - R;
629  residues.push_back(dca);
630  }
631  return residues;
632 }

634 std::vector<double> PHTpcTrackSeedVertexAssoc::GetLineClusterResiduals(std::vector<std::pair<double,double>> points, double A, double B)
635 {
636  std::vector<double> residues;
637  // calculate cluster residuals from the fitted circle
638  for(unsigned int i = 0; i < points.size(); ++i)
639  {
640  double r = points[i].first;
641  double z = points[i].second;
642 
643  // The shortest distance of a point from a circle is along the radial; line from the circle center to the point
644 
645  double a = -A;
646  double b = 1.0;
647  double c = -B;
648  double dca = sqrt(pow(a*r+b*z+c, 2)) / sqrt(a*a+b*b);
649 
650  residues.push_back(dca);
651  }
652  return residues;
