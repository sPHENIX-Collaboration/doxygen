680 void ALICEKF::repairCovariance(SvtxTrack_v2 &track)
681 {
682  // find closest positive definite matrix
683  Eigen::Matrix<double,6,6> cov = getEigenCov(track);
684  Eigen::SelfAdjointEigenSolver<Eigen::Matrix<double,6,6>> solver(cov);
685  Eigen::Matrix<double,6,1> D = solver.eigenvalues();
686  Eigen::Matrix<double,6,6> Q = solver.eigenvectors();
687  Eigen::Matrix<double,6,1> Dp = D.cwiseMax(1e-15);
688  Eigen::Matrix<double,6,6> Z = Q*Dp.asDiagonal()*Q.transpose();
689  // updates covariance matrix
690  for(int i=0;i<6;i++)
691  {
692  for(int j=0;j<6;j++)
693  {
694  track.set_error(i,j,Z(i,j));
695  }

698 void ALICEKF::CircleFitByTaubin (std::vector<std::pair<double,double>> points, double &R, double &X0, double &Y0)
699 /*
700  Circle fit to a given set of data points (in 2D)
701  This is an algebraic fit, due to Taubin, based on the journal article
702  G. Taubin, "Estimation Of Planar Curves, Surfaces And Nonplanar
703  Space Curves Defined By Implicit Equations, With
704  Applications To Edge And Range Image Segmentation",
705  IEEE Trans. PAMI, Vol. 13, pages 1115-1138, (1991)
706 */
707 {
708  int iter,IterMAX=99;
709 
710  double Mz,Mxy,Mxx,Myy,Mxz,Myz,Mzz,Cov_xy,Var_z;
711  double A0,A1,A2,A22,A3,A33;
712  double x,y;
713  double DET,Xcenter,Ycenter;
714 
715  // Compute x- and y- sample means
716  double meanX = 0;
717  double meanY = 0;
718  double weight = 0;
719  for(unsigned int i = 0; i < points.size(); ++i)
720  {
721  meanX += points[i].first;
722  meanY += points[i].second;
723  weight++;
724  }
725  meanX /= weight;
726  meanY /= weight;
727 
728  // computing moments
729 
730  Mxx=Myy=Mxy=Mxz=Myz=Mzz=0.;
731 
732  for (unsigned int i=0; i<points.size(); i++)
733  {
734  double Xi = points[i].first - meanX; // centered x-coordinates
735  double Yi = points[i].second - meanY; // centered y-coordinates
736  double Zi = Xi*Xi + Yi*Yi;
737 
738  Mxy += Xi*Yi;
739  Mxx += Xi*Xi;
740  Myy += Yi*Yi;
741  Mxz += Xi*Zi;
742  Myz += Yi*Zi;
743  Mzz += Zi*Zi;
744  }
745  Mxx /= weight;
746  Myy /= weight;
747  Mxy /= weight;
748  Mxz /= weight;
749  Myz /= weight;
750  Mzz /= weight;
751 
752  // computing coefficients of the characteristic polynomial
753 
754  Mz = Mxx + Myy;
755  Cov_xy = Mxx*Myy - Mxy*Mxy;
756  Var_z = Mzz - Mz*Mz;
757  A3 = 4*Mz;
758  A2 = -3*Mz*Mz - Mzz;
759  A1 = Var_z*Mz + 4*Cov_xy*Mz - Mxz*Mxz - Myz*Myz;
760  A0 = Mxz*(Mxz*Myy - Myz*Mxy) + Myz*(Myz*Mxx - Mxz*Mxy) - Var_z*Cov_xy;
761  A22 = A2 + A2;
762  A33 = A3 + A3 + A3;
763 
764  // finding the root of the characteristic polynomial
765  // using Newton's method starting at x=0
766  // (it is guaranteed to converge to the right root)
767 
768  for (x=0.,y=A0,iter=0; iter<IterMAX; iter++) // usually, 4-6 iterations are enough
769  {
770  double Dy = A1 + x*(A22 + A33*x);
771  double xnew = x - y/Dy;
772  if ((xnew == x)||(!isfinite(xnew))) break;
773  double ynew = A0 + xnew*(A1 + xnew*(A2 + xnew*A3));
774  if (fabs(ynew)>=fabs(y)) break;
775  x = xnew; y = ynew;
776  }
777 
778  // computing parameters of the fitting circle
779 
780  DET = x*x - x*Mz + Cov_xy;
781  Xcenter = (Mxz*(Myy - x) - Myz*Mxy)/DET/2;
782  Ycenter = (Myz*(Mxx - x) - Mxz*Mxy)/DET/2;
783 
784  // assembling the output
785 
786  X0 = Xcenter + meanX;
787  Y0 = Ycenter + meanY;
788  R = sqrt(Xcenter*Xcenter + Ycenter*Ycenter + Mz);
789 }

791 void ALICEKF::line_fit(std::vector<std::pair<double,double>> points, double &a, double &b)
792 {
793  // copied from: https://www.bragitoff.com
794  // we want to fit z vs radius
795 
796  double xsum=0,x2sum=0,ysum=0,xysum=0; //variables for sums/sigma of xi,yi,xi^2,xiyi etc
797  for (unsigned int i=0; i<points.size(); ++i)
798  {
799  double r = points[i].first;
800  double z = points[i].second;
801 
802  xsum=xsum+r; //calculate sigma(xi)
803  ysum=ysum+z; //calculate sigma(yi)
804  x2sum=x2sum+pow(r,2); //calculate sigma(x^2i)
805  xysum=xysum+r*z; //calculate sigma(xi*yi)
806  }
807  a=(points.size()*xysum-xsum*ysum)/(points.size()*x2sum-xsum*xsum); //calculate slope
808  b=(x2sum*ysum-xsum*xysum)/(x2sum*points.size()-xsum*xsum); //calculate intercept
809 
810  if(Verbosity() > 10)
811  {
812  for (unsigned int i=0;i<points.size(); ++i)
813  {
814  double r = points[i].first;
815  double z_fit = a * r + b; //to calculate z(fitted) at given r points
816  std::cout << " r " << r << " z " << points[i].second << " z_fit " << z_fit << std::endl;
817  }
818  }
819 
820  return;
821 }

840 std::vector<double> ALICEKF::GetCircleClusterResiduals(std::vector<std::pair<double,double>> points, double R, double X0, double Y0)
841 {

843  // calculate cluster residuals from the fitted circle
844  for(unsigned int i = 0; i < points.size(); ++i)
845  {
846  double x = points[i].first;
847  double y = points[i].second;
848 
849  // The shortest distance of a point from a circle is along the radial; line from the circle center to the point

851  residues.push_back(dca);
852  }
