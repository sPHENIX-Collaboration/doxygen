73  template< float (PHG4Hit::*accessor)(int) const>
74  float interpolate( std::set<PHG4Hit*> hits, float rextrap )
75  {
76  // calculate all terms needed for the interpolation
77  // need to use double everywhere here due to numerical divergences
78  double sw = 0;
79  double swr = 0;
80  double swr2 = 0;
81  double swx = 0;
82  double swrx = 0;
83 
84  bool valid( false );
85  for( const auto& hit:hits )
86  {
87 
88  const double x0 = (hit->*accessor)(0);
89  const double x1 = (hit->*accessor)(1);
90  if( std::isnan( x0 ) || std::isnan( x1 ) ) continue;
91 
92  const double w = hit->get_edep();
93  if( w < 0 ) continue;
94 
95  valid = true;
96  const double r0 = get_r( hit, 0 );
97  const double r1 = get_r( hit, 1 );
98 
99  sw += w*2;
100  swr += w*(r0 + r1);
101  swr2 += w*(square(r0) + square(r1));
102  swx += w*(x0 + x1);
103  swrx += w*(r0*x0 + r1*x1);
104  }
105 
106  if( !valid ) return NAN;
107 
108  const auto alpha = (sw*swrx - swr*swx);
109  const auto beta = (swr2*swx - swr*swrx);
110  const auto denom = (sw*swr2 - square(swr));
111 
112  return ( alpha*rextrap + beta )/denom;
113  }

184  void add_trk_information( TrackEvaluationContainerv1::ClusterStruct& cluster, SvtxTrackState* state )
185  {
186  // need to extrapolate the track state to the right cluster radius to get proper residuals
187  const auto trk_r = get_r( state->get_x(), state->get_y() );
188  const auto dr = cluster.r - trk_r;
189  const auto trk_drdt = (state->get_x()*state->get_px() + state->get_y()*state->get_py())/trk_r;
190  const auto trk_dxdr = state->get_px()/trk_drdt;
191  const auto trk_dydr = state->get_py()/trk_drdt;
192  const auto trk_dzdr = state->get_pz()/trk_drdt;
193 
194  // store state position
195  cluster.trk_x = state->get_x() + dr*trk_dxdr;
196  cluster.trk_y = state->get_y() + dr*trk_dydr;
197  cluster.trk_z = state->get_z() + dr*trk_dzdr;
198  cluster.trk_r = get_r( cluster.trk_x, cluster.trk_y );
199  cluster.trk_phi = std::atan2( cluster.trk_y, cluster.trk_x );
200 
201  /* store local momentum information */
202  cluster.trk_px = state->get_px();
203  cluster.trk_py = state->get_py();
204  cluster.trk_pz = state->get_pz();
205 
206  /*
207  store state angles in (r,phi) and (r,z) plans
208  they are needed to study space charge distortions
209  */
210  const auto cosphi( std::cos( cluster.trk_phi ) );
211  const auto sinphi( std::sin( cluster.trk_phi ) );
212  const auto trk_pphi = -state->get_px()*sinphi + state->get_py()*cosphi;
213  const auto trk_pr = state->get_px()*cosphi + state->get_py()*sinphi;
214  const auto trk_pz = state->get_pz();
215  cluster.trk_alpha = std::atan2( trk_pphi, trk_pr );
216  cluster.trk_beta = std::atan2( trk_pz, trk_pr );

310  // add truth information
311  void add_truth_information( TrackEvaluationContainerv1::ClusterStruct& cluster, std::set<PHG4Hit*> hits )
312  {
313  // need to extrapolate g4hits position to the cluster r
314  /* this is done by using a linear extrapolation, with a straight line going through all provided G4Hits in and out positions */
315  const auto rextrap = cluster.r;
316  cluster.truth_size = hits.size();
317  cluster.truth_x = interpolate<&PHG4Hit::get_x>( hits, rextrap );
318  cluster.truth_y = interpolate<&PHG4Hit::get_y>( hits, rextrap );
319  cluster.truth_z = interpolate<&PHG4Hit::get_z>( hits, rextrap );
320  cluster.truth_r = get_r( cluster.truth_x, cluster.truth_y );
321  cluster.truth_phi = std::atan2( cluster.truth_y, cluster.truth_x );
322 
323  /* add truth momentum information */
324  cluster.truth_px = interpolate<&PHG4Hit::get_px>( hits, rextrap );
325  cluster.truth_py = interpolate<&PHG4Hit::get_py>( hits, rextrap );
326  cluster.truth_pz = interpolate<&PHG4Hit::get_pz>( hits, rextrap );
327 
328  /*
329  store state angles in (r,phi) and (r,z) plans
330  they are needed to study space charge distortions
331  */
332  const auto cosphi( std::cos( cluster.truth_phi ) );
333  const auto sinphi( std::sin( cluster.truth_phi ) );
334  const auto truth_pphi = -cluster.truth_px*sinphi + cluster.truth_py*cosphi;
335  const auto truth_pr = cluster.truth_px*cosphi + cluster.truth_py*sinphi;
336 
337  cluster.truth_alpha = std::atan2( truth_pphi, truth_pr );
338  cluster.truth_beta = std::atan2( cluster.truth_pz, truth_pr );
339  if(std::isnan(cluster.truth_alpha) || std::isnan(cluster.truth_beta))
340  {
341  // recalculate
342  double truth_alpha = 0;
343  double truth_beta = 0;
344  double sum_w = 0;
345  for( const auto& hit:hits )
346  {
347  const auto px = hit->get_x(1) - hit->get_x(0);
348  const auto py = hit->get_y(1) - hit->get_y(0);
349  const auto pz = hit->get_z(1) - hit->get_z(0);
350  const auto pphi = -px*sinphi + py*cosphi;
351  const auto pr = px*cosphi + py*sinphi;
352 
353  const auto w = hit->get_edep();
354  if( w < 0 ) continue;
355 
356  sum_w += w;
357  truth_alpha += w*std::atan2( pphi, pr );
358  truth_beta += w*std::atan2( pz, pr );
359  }
360  truth_alpha /= sum_w;
361  truth_beta /= sum_w;
362  cluster.truth_alpha = truth_alpha;
363  cluster.truth_beta = truth_beta;
364  }

602  /* this assumes that both clusters and states are sorted along r */
603  const auto radius( cluster_struct.r );
604  float dr_min = -1;
605  for( auto iter = state_iter; iter != track->end_states(); ++iter )
606  {
607  const auto dr = std::abs( radius - get_r( iter->second->get_x(), iter->second->get_y() ) );
608  if( dr_min < 0 || dr < dr_min )
609  {
610  state_iter = iter;
611  dr_min = dr;
612  } else break;
613  }
614 
615  // store track state in cluster struct
616  add_trk_information( cluster_struct, state_iter->second );
617 
618  // add to track
619  track_struct.clusters.push_back( cluster_struct );
620  }
621  m_container->addTrack( track_struct );
622  }
623 
624  std::cout << "TrackEvaluation::evaluate_tracks - tracks: " << m_container->tracks().size() << std::endl;
625

630 TrackEvaluation::G4HitSet TrackEvaluation::find_g4hits( TrkrDefs::cluskey cluster_key ) const
631 {
632 
633  // check maps
634  if( !( m_cluster_hit_map && m_hit_truth_map ) ) return G4HitSet();
635 
636  // check if in map
637  auto map_iter = m_g4hit_map.lower_bound( cluster_key );
638  if( map_iter != m_g4hit_map.end() && cluster_key == map_iter->first )
639  { return map_iter->second; }
640 
641  // find hitset associated to cluster
642  G4HitSet out;
643  const auto hitset_key = TrkrDefs::getHitSetKeyFromClusKey(cluster_key);
644  for(const auto& [first,hit_key]:range_adaptor( m_cluster_hit_map->getHits(cluster_key)))
645  {
646 
647  // store hits to g4hit associations
648  TrkrHitTruthAssoc::MMap g4hit_map;
649  m_hit_truth_map->getG4Hits( hitset_key, hit_key, g4hit_map );
650 
651  // find corresponding g4 hist
652  for( auto truth_iter = g4hit_map.begin(); truth_iter != g4hit_map.end(); ++truth_iter )
653  {
654 
655  const auto g4hit_key = truth_iter->second.second;
656  PHG4Hit* g4hit = nullptr;
657 
658  switch( TrkrDefs::getTrkrId( hitset_key ) )
659  {
660  case TrkrDefs::mvtxId:
661  if( m_g4hits_mvtx ) g4hit = m_g4hits_mvtx->findHit( g4hit_key );
662  break;
663 
664  case TrkrDefs::inttId:
665  if( m_g4hits_intt ) g4hit = m_g4hits_intt->findHit( g4hit_key );
666  break;
667 
668  case TrkrDefs::tpcId:
669  if( m_g4hits_tpc ) g4hit = m_g4hits_tpc->findHit( g4hit_key );
670  break;
671 
672  case TrkrDefs::micromegasId:
673  if( m_g4hits_micromegas ) g4hit = m_g4hits_micromegas->findHit( g4hit_key );
674  break;
675 
676  default: break;
677  }
678 
679  if( g4hit ) out.insert( g4hit );
680  else std::cout << "TrackEvaluation::find_g4hits - g4hit not found " << g4hit_key << std::endl;
681 
682  }
683  }
684 
685  // insert in map and return
686  return m_g4hit_map.insert( map_iter, std::make_pair( cluster_key, std::move( out ) ) )->second;
687

691 std::pair<int,int> TrackEvaluation::get_max_contributor( SvtxTrack* track ) const
692 {
693  if(!(m_track_map && m_cluster_map && m_g4truthinfo)) return {0,0};
694 
695  // maps MC track id and number of matching g4hits
696  using IdMap = std::map<int,int>;
697  IdMap contributor_map;
698 
699  // loop over clusters
700  for( auto key_iter = track->begin_cluster_keys(); key_iter != track->end_cluster_keys(); ++key_iter )
701  {
702  const auto& cluster_key = *key_iter;
703  for( const auto& hit:find_g4hits( cluster_key ) )
704  {
705  const int trkid = hit->get_trkid();
706  auto iter = contributor_map.lower_bound( trkid );
707  if( iter == contributor_map.end() || iter->first != trkid )
708  {
709  contributor_map.insert(iter, std::make_pair(trkid,1));
710  } else ++iter->second;
711  }
712  }
713 
714  if( contributor_map.empty() ) return {0,0};
715  else return *std::max_element(
716  contributor_map.cbegin(), contributor_map.cend(),
717  []( const IdMap::value_type& first, const IdMap::value_type& second )
718  { return first.second < second.second; } );
719 
720 }
