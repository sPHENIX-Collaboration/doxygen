22 void Acts::AccumulatedMaterialSlab::trackVariance(MaterialSlab slabReference,
23  bool useEmptyTrack) {
24  // Only use real tracks or if empty tracks are allowed.
25  if (useEmptyTrack or (0 < m_trackAverage.thickness())) {
26  float variance = ((1 / m_trackAverage.material().X0()) -
27  (1 / slabReference.material().X0())) *
28  ((1 / m_trackAverage.material().X0()) -
29  (1 / slabReference.material().X0()));
30  if (m_totalCount == 0u) {
31  m_totalVariance = variance;
32  } else {
33  double weightTotal = m_totalCount / (m_totalCount + 1.0);
34  double weightTrack = 1 / (m_totalCount + 1.0);
35  m_totalVariance = weightTotal * m_totalVariance + weightTrack * variance;
36  }
37  }
38 }

40 void Acts::AccumulatedMaterialSlab::trackAverage(bool useEmptyTrack) {
41  // average only real tracks or if empty tracks are allowed.
42  if (useEmptyTrack or (0 < m_trackAverage.thickness())) {
43  if (m_totalCount == 0u) {
44  m_totalAverage = m_trackAverage;
45  } else {
46  double weightTotal = m_totalCount / (m_totalCount + 1.0);
47  double weightTrack = 1 / (m_totalCount + 1.0);
48  // average such that each track contributes equally.
49  MaterialSlab fromTotal(m_totalAverage.material(),
50  weightTotal * m_totalAverage.thickness());
51  MaterialSlab fromTrack(m_trackAverage.material(),
52  weightTrack * m_trackAverage.thickness());
53  m_totalAverage = detail::combineSlabs(fromTotal, fromTrack);
54  }
55  m_totalCount += 1;
56  }
57  // reset track average
58  m_trackAverage = MaterialSlab();
59 }
