33 std::vector<std::string> tokenize(const std::string& input)
34 {
35  typedef boost::escaped_list_separator<char> separator_type;
36  separator_type separator("\\", // The escape characters.
37  "= ", // The separator characters.
38  "\"\'"); // The quote characters.
39 
40  // Tokenize the intput.
41  boost::tokenizer<separator_type> tokens(input, separator);
42 
43  // Copy non-empty tokens from the tokenizer into the result.
44  std::vector<std::string> result;
45  copy_if(tokens.begin(), tokens.end(), std::back_inserter(result),
46  !boost::bind(&std::string::empty, _1));
47  return result;
48 }

210 void TrentoInitial::PreDefined(std::string stored_system,
211  double centrality_low, double centrality_high,
212  double grid_max, double grid_step, unsigned random_seed)
213 {
214  VarMap var_map{};
215  if (stored_system == "auau200") {
216  std::string projectile = "Au";
217  std::string target = "Au";
218  double cross_section = 4.23;
219  var_map = create_varmap(projectile, target, cross_section,
220  grid_max, grid_step, random_seed);
221  } else if (stored_system == "pbpb2760") {
222  std::string projectile = "Pb";
223  std::string target = "Pb";
224  double cross_section = 6.4;
225  var_map = create_varmap(projectile, target, cross_section,
226  grid_max, grid_step, random_seed);
227  } else if (stored_system == "pbpb5020") {
228  std::string projectile = "Pb";
229  std::string target = "Pb";
230  double cross_section = 7.0;
231  var_map = create_varmap(projectile, target, cross_section,
232  grid_max, grid_step, random_seed);
233  }
234 
235  TrentoCollision collision_(var_map);
236 
237  double smin, smax;
238  std::tie(smin, smax) = GetEntropyRange(stored_system,
239  centrality_low, centrality_high);
240  collision_.Sample(smin, smax);
241  info_ = collision_.info_;
242  for (const auto& row : collision_.event_.reduced_thickness_grid()) {
243  auto&& iter = row.begin();
244  // Write all row elements except the last with a space delimiter afterwards.
245  do {
246  entropy_density_distribution_.push_back(*iter);
247  } while (++iter != --row.end());
248  // Write the last element and a linebreak.
249  entropy_density_distribution_.push_back(*iter);
250  }
251 
252  int nx = int(std::sqrt(entropy_density_distribution_.size()));
253  double xmax = nx * grid_step / 2;
254  SetRanges(xmax, xmax, 0.0);
255  SetSteps(grid_step, grid_step, 0.0);
256  ComputeNbc();
257 }

297 void TrentoInitial::Exec() {
298  VERBOSE(2) << " : Excute initial condition ";
299  if (!trento_xml_) {
300  WARN << " : Not a valid JetScape IS::Trento XML section in file!";
301  exit(-1);
302  } else {
303  // trento_xml_->Attribute("A", "B") checks whether the attribute "A" has value "B"
304  auto random_seed = (*GetMt19937Generator())();
305  VERBOSE(2) << "Random seed used for TrentoInitial class" << random_seed;
306  if ( trento_xml_->Attribute("use_module", "pre_defined") ) {
307  auto predef = trento_xml_->FirstChildElement("pre_defined");
308  std::string collision_system(predef->Attribute("collision_system"));
309  VERBOSE(2) << "collision_system=" << collision_system;
310  double centrality_min = std::atof(predef->Attribute("centrality_min"));
311  double centrality_max = std::atof(predef->Attribute("centrality_max"));
312  PreDefined(collision_system, centrality_min, centrality_max,
313  GetXMax(), GetXStep(), random_seed);
314  } else if (trento_xml_->Attribute("use_module", "user_defined") ) {
315  auto usrdef = trento_xml_->FirstChildElement("user_defined");
316  std::string projectile(usrdef->Attribute("projectile"));
317  std::string target(usrdef->Attribute("target"));
318  // center of mass collision energy per pair of nucleon
319  double sqrts_NN = std::atof(usrdef->Attribute("sqrts"));
320  double cross_section = std::atof(usrdef->Attribute("cross_section"));

343 std::tuple<double, double> TrentoInitial::GetEntropyRange(std::string collision_system,
344  double centrality_low, double centrality_high) {
345  std::stringstream centrality_class_path;
346  centrality_class_path << "data_table/trento_" << collision_system
347  << "_cent.csv";
348 
349  std::ifstream fin(centrality_class_path.str());
350  if (!fin.is_open()) {
351  throw runtime_error("open "
352  + centrality_class_path.str() + " failed");
353  }
354  char buf[256];
355  fin.getline(buf, 256);
356 
357  double centrality_bound[101];
358  double entropy_bound[101];
359 
360  for (int i = 0; i < 101; i++) {
361  fin >> centrality_bound[i] >> entropy_bound[i];
362  }
363 
364  fin.close();
365 
366  auto interp1d = [&](double cent) {
367  if (cent < 0.0 || cent > 100.0) {
368  throw RangeFailure(std::string("centrality ") + std::to_string(cent)
369  + " is not in the range [0, 100]");
370  } else {
371  double x0, x1, y0, y1;
372  int j = int(cent);
373  if (j > cent) {
374  x0 = centrality_bound[j-1];
375  y0 = entropy_bound[j-1];
376  x1 = centrality_bound[j];
377  y1 = entropy_bound[j];
378  } else {
379  x0 = centrality_bound[j];
380  y0 = entropy_bound[j];
381  x1 = centrality_bound[j+1];
382  y1 = entropy_bound[j+1];
383  }
384  double r = (cent - x0)/(x1 - x0);
385  return (1.0 - r) * y0 + r * y1;
386  }
387  };
388 
389  return std::make_pair(interp1d(centrality_high),
390  interp1d(centrality_low));
391 }
