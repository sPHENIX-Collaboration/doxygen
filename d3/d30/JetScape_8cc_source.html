132 void JetScape::Exec()
133 {
134  INFO<<BOLDRED<<"Run JetScape ...";
135  INFO<<BOLDRED<<"Number of Events = "<<GetNumberOfEvents();
136 
137  // JetScapeTask::ExecuteTasks(); Has to be called explicitly since not really fully recursively (if ever needed)
138  // --> JetScape is "Task Manager" of all modules ...
139 
140  // Simple way of passing the writer module pointer
141  vector<weak_ptr<JetScapeWriter>> vWriter;
142 
143  for (auto it : GetTaskList()) {
144  if (dynamic_pointer_cast<JetScapeWriter>(it)){
145  if (it->GetActive())
146  vWriter.push_back(dynamic_pointer_cast<JetScapeWriter>(it));
147  }
148  }
149 
150  for (int i=0;i<GetNumberOfEvents();i++)
151  {
152  INFO<<BOLDRED<<"Run Event # = "<<i;
153  JSDEBUG<<"Found "<<GetNumberOfTasks()<<" Modules Execute them ... ";
154 
155  // First run all tasks
156  JetScapeTask::ExecuteTasks();
157 
158  // Then hand around the collection of writers and ask
159  // modules to write what they like
160  // Sequence of events:
161  // -- writer->Exec is called and redirects to WriteEvent, which starts a new event line
162  // -- any remaining exec's finish
163  // -- all modules write their headers
164  // -- Now all header info is known to the writers, so write out the header
165  // -- all other Write()'s are being called
166  // the result still confuses me. It's in the best possible order but it shouldn't be.
167 
168  // collect module header data
169  for (auto w : vWriter) {
170  auto f = w.lock();
171  if ( f ) JetScapeTask::CollectHeaders(w);
172  }
173  // official header
174  for (auto w : vWriter) {
175  auto f = w.lock();
176  if ( f ) f->WriteHeaderToFile();
177  }
178  // event data
179  for (auto w : vWriter) {
180  auto f = w.lock();
181  if ( f ) JetScapeTask::WriteTasks(w);
182  }
183 
184  // Finalize
185  for (auto w : vWriter) {
186  auto f = w.lock();
187  if ( f ) f->WriteEvent();
188  }
189 
190  // For reusal, deactivate task after it has finished but before it gets cleaned up.
191  if (reuse_hydro_) {
192  if (n_reuse_hydro_ <= 0) {
193  WARN << " reuse_hydro is set, but n_reuse_hydro=" << n_reuse_hydro_;
194  throw std::runtime_error ("Incompatible reusal settings.");
195  }
196  for (auto it : GetTaskList()) {
197  if (!dynamic_pointer_cast<FluidDynamics>(it)
198  && !dynamic_pointer_cast<InitialState>(it)) {
199  continue;
200  }
201  if (i%n_reuse_hydro_ == n_reuse_hydro_ - 1) {
202  JSDEBUG << " i was " << i << " i%n_reuse_hydro_ = " << i%n_reuse_hydro_ << " --> ACTIVATING";
203  it->SetActive(true);
204  } else {
205  JSDEBUG << " i was " << i << " i%n_reuse_hydro_ = " << i%n_reuse_hydro_ << " --> DE-ACTIVATING";
206  it->SetActive(false);
207  }
208  }
209  }
210  // Now clean up, only affects active taskjs
211  JetScapeTask::ClearTasks();
212 
213  IncrementCurrentEvent();
214  }
215 }
