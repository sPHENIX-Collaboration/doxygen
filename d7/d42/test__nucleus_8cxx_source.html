196 TEST_CASE( "woods-saxon sampling" ) {
197  int A = 200;
198  double R = 6., a = .5;
199  NucleusPtr nucleus{new WoodsSaxonNucleus{static_cast<std::size_t>(A), R, a}};
200 
201  // Test Woods-Saxon sampling.
202  // This is honestly not a great test; while it does prove that the code
203  // basically works as intended, it does not rigorously show that the generated
204  // numbers are actually Woods-Saxon distributed. Personally I am much more
205  // convinced by plotting a histogram and visually comparing it to the smooth
206  // curve. The script 'plot-woods-saxon.py' in the 'woods-saxon' subdirectory
207  // does this.
208 
209  // sample a bunch of nuclei and bin all the nucleon positions
210  // using "p" for cylindrical radius to distinguish from spherical "r"
211  auto dp = .5;
212  auto nevents = 1000;
213  auto nsamples = nevents * A;
214  std::map<int, int> hist{};
215  for (auto i = 0; i < nevents; ++i) {
216  nucleus->sample_nucleons(0.);
217  for (const auto& nucleon : *nucleus) {
218  auto x = nucleon.x();
219  auto y = nucleon.y();
220  auto p = std::sqrt(x*x + y*y);
221  ++hist[p/dp];
222  }
223  }
224 
225  // integrate the W-S dist from pmin to pmax and over all z
226  double rmax = R + 10.*a;
227  auto integrate_woods_saxon = [R, a, rmax](double pmin, double pmax) -> double {
228  int nzbins = 1000;
229  auto npbins = static_cast<int>(nzbins*(pmax-pmin)/rmax);
230  double result = 0.;
231  for (auto ip = 0; ip <= npbins; ++ip) {
232  auto p = pmin + (ip*(pmax-pmin))/(npbins - 1);
233  for (auto iz = 0; iz < nzbins; ++iz) {
234  auto z = (iz*rmax)/(nzbins-1);
235  result += p/(1.+std::exp((std::sqrt(p*p + z*z) - R)/a));
236  }
237  }
238  return result;
239  };
240 
241  double ws_norm = integrate_woods_saxon(0, rmax);
242 
243  // check all histogram bins agree with numerical integration
244  auto all_bins_correct = true;
245  std::ostringstream bin_output{};
246  bin_output << std::fixed << std::boolalpha
247  << "pmin pmax prob cprob ratio pass\n";
248  for (const auto& bin : hist) {
249  auto pmin = bin.first * dp;
250  auto pmax = pmin + dp;
251  auto prob = static_cast<double>(bin.second) / nsamples;
252  auto correct_prob = integrate_woods_saxon(pmin, pmax) / ws_norm;
253  bool within_tol = prob == Approx(correct_prob).epsilon(.1);
254  if (!within_tol)
255  all_bins_correct = false;
256  bin_output << pmin << ' '
257  << pmax << ' '
258  << prob << ' '
259  << correct_prob << ' '
260  << prob/correct_prob << ' '
261  << within_tol << '\n';
262  }
263  INFO( bin_output.str() );
264  CHECK( all_bins_correct );
265 }

267 TEST_CASE( "deformed woods-saxon sampling" ) {
268  // This is tough to test because the sampled positions are randomly rotated
269  // and the z-coordinate is discarded. However, the authors have visually
270  // checked the results by histogramming the samples (not done here, but easy
271  // to reproduce).
272 
273  // As a not-very-stringent test, check that the mean ellipticity of a deformed
274  // W-S nucleus is significantly larger than a symmetric nucleus.
275 
276  std::size_t A = 200;
277  double R = 6., a = .5, beta2 = .3, beta4 = .1;
278 
279  auto nucleus_sym = WoodsSaxonNucleus{A, R, a};
280  auto nucleus_def = DeformedWoodsSaxonNucleus{A, R, a, beta2, beta4};
281 
282  auto mean_ecc2 = [](Nucleus& nucleus) {
283  double e2 = 0.;
284  int nevents = 500;
285 
286  for (int n = 0; n < nevents; ++n) {
287  nucleus.sample_nucleons(0.);
288  double numer = 0.;
289  double denom = 0.;
290  for (const auto& nucleon : nucleus) {
291  double x2 = std::pow(nucleon.x(), 2);
292  double y2 = std::pow(nucleon.y(), 2);
293  numer += x2 - y2;
294  denom += x2 + y2;
295  }
296 
297  e2 += std::fabs(numer) / denom;
298  }
299 
300  return e2 / nevents;
301  };
302 
303  CHECK( mean_ecc2(nucleus_def) > 2.*mean_ecc2(nucleus_sym) );
304 }
