683 void JetBkgdSub::EstimateRhoMult(PHCompositeNode *topNode)
684 {
685 
686  //==================================
687  // Estimate rho
688  //==================================
689  m_rho_mult = -1;
690 
691  // iternstruct kT jets
692  std::vector<Jet *> inputs; // owns memory
693  for (auto & _input : _inputs)
694  {
695  std::vector<Jet *> parts = _input->get_input(topNode);
696  for (auto & part : parts)
697  {
698  inputs.push_back(part);
699  inputs.back()->set_id(inputs.size() - 1); // unique ids ensured

702 
703  // define kT jet algorithm

714  std::vector<double> pt_over_nConstituents;
715  // loop over kT jets
716  int nktjets = 0;
717  for(unsigned int iktjet = 0; iktjet < kt_jets.size(); iktjet++)
718  {
719  if (iktjet <= 1) continue; // skip the two leading jets

727  }

732  return;
