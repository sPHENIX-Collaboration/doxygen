43  m_cfg.gammaParameters[1]);
44  std::poisson_distribution<int> poissonDist(m_cfg.poissonParameter);
45 
46  ACTS_INFO(m_cfg.drawsPerEvent << " draws per event will be done");
47 
48  // generate collection of random numbers
49  HelloDataCollection collection;
50  for (size_t idraw = 0; idraw < m_cfg.drawsPerEvent; ++idraw) {
51  double gauss = gaussDist(rng);
52  double uniform = uniformDist(rng);
53  double gamma = gammaDist(rng);
54  int poisson = poissonDist(rng);
55 
56  ACTS_VERBOSE("Gauss : " << gauss);
57  ACTS_VERBOSE("Uniform : " << uniform);
58  ACTS_VERBOSE("Gamma : " << gamma);
59  ACTS_VERBOSE("Poisson : " << poisson);
60 
61  HelloData x{};

63  x.a = uniform;
64  x.b = gamma;
65  x.t = poisson;
66  collection.push_back(x);
67  }
68 
69  // transfer generated data to the event store.
70  m_writeHandle(ctx, std::move(collection));

72  return ActsExamples::ProcessCode::SUCCESS;
