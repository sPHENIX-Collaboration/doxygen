29 BOOST_AUTO_TEST_CASE(WithoutInteraction) {
30  // some hit position
31  auto p4 = Hit::Vector4(1, 2, 3, 4);
32  // before/after four-momenta are the same
33  auto m4 = Hit::Vector4(1, 1, 1, 4);
34  auto h = Hit(gid, pid, p4, m4, m4, 12u);
35 
36  BOOST_CHECK_EQUAL(h.geometryId(), gid);
37  BOOST_CHECK_EQUAL(h.particleId(), pid);
38  BOOST_CHECK_EQUAL(h.index(), 12u);
39  CHECK_CLOSE_REL(h.fourPosition(), p4, eps);
40  CHECK_CLOSE_REL(h.position(), Hit::Vector3(1, 2, 3), eps);
41  CHECK_CLOSE_REL(h.time(), 4, eps);
42  CHECK_CLOSE_REL(h.momentum4Before(), m4, eps);
43  CHECK_CLOSE_REL(h.momentum4After(), m4, eps);
44  CHECK_CLOSE_REL(h.directionBefore(), Hit::Vector3(1, 1, 1).normalized(), eps);
45  CHECK_CLOSE_REL(h.directionAfter(), Hit::Vector3(1, 1, 1).normalized(), eps);
46  CHECK_CLOSE_REL(h.direction(), Hit::Vector3(1, 1, 1).normalized(), eps);
