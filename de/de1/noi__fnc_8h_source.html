142 
143  TH1D* ret;

150  for (int j=1;j<den->GetNbinsX()+1;++j){

152  double d = den->GetBinContent(j) / norm_den;
153  if (n == 0 || d == 0) {
154  ret->SetBinContent(j,0);
155  ret->SetBinError(j,0);

157  continue;
158  }

160  double d_err = den->GetBinError(j) / norm_den;
161  double val = n / d;

164  ret->SetBinError(j,err);
165  }
