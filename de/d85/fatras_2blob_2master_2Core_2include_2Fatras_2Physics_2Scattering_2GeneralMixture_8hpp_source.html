104  std::array<double, 4> getGaussian(double beta, double p, double tInX0,
105  double scale) const {
106  std::array<double, 4> scattering_params;
107  // Total standard deviation of mixture
108  scattering_params[0] = 15. / beta / p * std::sqrt(tInX0) * scale;
109  scattering_params[1] = 1.0; // Variance of core
110  scattering_params[2] = 1.0; // Variance of tails
111  scattering_params[3] = 0.5; // Mixture weight of tail component

115  std::array<double, 4> getGaussmix(double beta, double p, double tInX0,
116  double Z, double scale) const {
117  std::array<double, 4> scattering_params;
118  scattering_params[0] = 15. / beta / p * std::sqrt(tInX0) *
119  scale; // Total standard deviation of mixture
120  double d1 = std::log(tInX0 / (beta * beta));
121  double d2 = std::log(std::pow(Z, 2.0 / 3.0) * tInX0 / (beta * beta));
122  double epsi;
123  double var1 = (-1.843e-3 * d1 + 3.347e-2) * d1 + 8.471e-1; // Variance of
124  // core
125  if (d2 < 0.5)
126  epsi = (6.096e-4 * d2 + 6.348e-3) * d2 + 4.841e-2;
127  else
128  epsi = (-5.729e-3 * d2 + 1.106e-1) * d2 - 1.908e-2;
129  scattering_params[1] = var1; // Variance of core
130  scattering_params[2] = (1 - (1 - epsi) * var1) / epsi; // Variance of tails
131  scattering_params[3] = epsi; // Mixture weight of tail component
132  return scattering_params;
133  }

135  std::array<double, 6> getSemigauss(double beta, double p, double tInX0,
136  double Z, double scale) const {
137  std::array<double, 6> scattering_params;
138  double N = tInX0 * 1.587E7 * std::pow(Z, 1.0 / 3.0) / (beta * beta) /
139  (Z + 1) / std::log(287 / std::sqrt(Z));
140  scattering_params[4] = 15. / beta / p * std::sqrt(tInX0) *
141  scale; // Total standard deviation of mixture
142  double rho = 41000 / std::pow(Z, 2.0 / 3.0);
143  double b = rho / std::sqrt(N * (std::log(rho) - 0.5));
144  double n = std::pow(Z, 0.1) * std::log(N);
145  double var1 = (5.783E-4 * n + 3.803E-2) * n + 1.827E-1;
146  double a =
147  (((-4.590E-5 * n + 1.330E-3) * n - 1.355E-2) * n + 9.828E-2) * n +
148  2.822E-1;
149  double epsi = (1 - var1) / (a * a * (std::log(b / a) - 0.5) - var1);
150  scattering_params[3] =
151  (epsi > 0) ? epsi : 0.0; // Mixture weight of tail component
152  scattering_params[0] = a; // Parameter 1 of tails
153  scattering_params[1] = b; // Parameter 2 of tails
154  scattering_params[2] = var1; // Variance of core
155  scattering_params[5] = N; // Average number of scattering processes
156  return scattering_params;
157  }

167  template <typename generator_t>
168  double gaussmix(UniformDist &udist, generator_t &generator,
169  const std::array<double, 4> &scattering_params) const {
170  double sigma_tot = scattering_params[0];
171  double var1 = scattering_params[1];
172  double var2 = scattering_params[2];
173  double epsi = scattering_params[3];
174  bool ind = udist(generator) > epsi;
175  double u = udist(generator);
176  if (ind)
177  return std::sqrt(var1) * std::sqrt(-2 * std::log(u)) * sigma_tot;
178  else
179  return std::sqrt(var2) * std::sqrt(-2 * std::log(u)) * sigma_tot;
180  }

190  template <typename generator_t>
191  double semigauss(UniformDist &udist, generator_t &generator,
192  const std::array<double, 6> &scattering_params) const {
193  double a = scattering_params[0];
194  double b = scattering_params[1];
195  double var1 = scattering_params[2];
196  double epsi = scattering_params[3];
197  double sigma_tot = scattering_params[4];
198  bool ind = udist(generator) > epsi;
199  double u = udist(generator);
200  if (ind)
201  return std::sqrt(var1) * std::sqrt(-2 * std::log(u)) * sigma_tot;
202  else
203  return a * b * std::sqrt((1 - u) / (u * b * b + a * a)) * sigma_tot;
204  }
