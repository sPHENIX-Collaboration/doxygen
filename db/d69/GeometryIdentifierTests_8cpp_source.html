37 BOOST_AUTO_TEST_CASE(GeometryIdentifier_max_values) {
38  // compute maximum value for each component
39  constexpr GeometryIdentifier::Value volumeMax = (1u << 8) - 1;
40  constexpr GeometryIdentifier::Value boundaryMax = (1u << 8) - 1;
41  constexpr GeometryIdentifier::Value layerMax = (1u << 12) - 1;
42  constexpr GeometryIdentifier::Value approachMax = (1u << 8) - 1;
43  constexpr GeometryIdentifier::Value sensitiveMax = (1u << 20) - 1;
44  constexpr GeometryIdentifier::Value extraMax = (1u << 8) - 1;
45  // reference values non-zero values everywhere
46  constexpr GeometryIdentifier ref = 0xdeadaffe01234567;
47  // values above the maximum are truncated
48  // max+1 has all available bits zeroed
49  BOOST_CHECK_EQUAL(GeometryIdentifier(ref).setVolume(volumeMax + 1),
50  GeometryIdentifier(ref).setVolume(0u));
51  BOOST_CHECK_EQUAL(GeometryIdentifier(ref).setBoundary(boundaryMax + 1),
52  GeometryIdentifier(ref).setBoundary(0u));
53  BOOST_CHECK_EQUAL(GeometryIdentifier(ref).setLayer(layerMax + 1),
54  GeometryIdentifier(ref).setLayer(0u));
55  BOOST_CHECK_EQUAL(GeometryIdentifier(ref).setApproach(approachMax + 1),
56  GeometryIdentifier(ref).setApproach(0u));
57  BOOST_CHECK_EQUAL(GeometryIdentifier(ref).setSensitive(sensitiveMax + 1),
58  GeometryIdentifier(ref).setSensitive(0u));
59  BOOST_CHECK_EQUAL(GeometryIdentifier(ref).setExtra(extraMax + 1),
60  GeometryIdentifier(ref).setExtra(0u));

63 BOOST_AUTO_TEST_CASE(GeometryIdentifier_order) {
64  auto vol1 = GeometryIdentifier()
65  .setVolume(1u)
66  .setLayer(14u)
67  .setSensitive(5u)
68  .setExtra(42u);
69  auto vol2 = GeometryIdentifier()
70  .setVolume(2u)
71  .setLayer(13u)
72  .setSensitive(3u)
73  .setExtra(43u);
74  // order uses volume first even if other components are larger
75  BOOST_CHECK_LT(vol1, vol2);
76  BOOST_CHECK_LT(GeometryIdentifier(vol1).setBoundary(64u), vol2);
77  BOOST_CHECK_LT(GeometryIdentifier(vol1).setLayer(64u), vol2);
78  BOOST_CHECK_LT(GeometryIdentifier(vol1).setApproach(64u), vol2);
79  BOOST_CHECK_LT(GeometryIdentifier(vol1).setSensitive(64u), vol2);
80  BOOST_CHECK_LT(GeometryIdentifier(vol1).setSensitive(64u), vol2);
81  BOOST_CHECK_LT(vol2, GeometryIdentifier(vol1).setVolume(3u));
82  // other components are hierarchical
83  BOOST_CHECK_LT(GeometryIdentifier(vol1).setVolume(1u).setBoundary(2u),
