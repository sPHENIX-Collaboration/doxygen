154 void TPC_Cells()
155 {
156  int verbosity = std::max(Enable::VERBOSITY, Enable::TPC_VERBOSITY);
157  Fun4AllServer* se = Fun4AllServer::instance();
158 
159  //=========================
160  // setup Tpc readout for filling cells
161  // g4tpc/PHG4TpcElectronDrift uses
162  // g4tpc/PHG4TpcPadPlaneReadout
163  //=========================
164 
165  PHG4TpcPadPlane* padplane = new PHG4TpcPadPlaneReadout();
166  padplane->Verbosity(verbosity);
167 
168  PHG4TpcElectronDrift* edrift = new PHG4TpcElectronDrift();
169  edrift->Detector("TPC");
170  edrift->Verbosity(verbosity);
171  if( G4TPC::ENABLE_STATIC_DISTORTIONS || G4TPC::ENABLE_TIME_ORDERED_DISTORTIONS )
172  {
173  auto distortionMap = new PHG4TpcDistortion;
174  distortionMap->set_do_static_distortions( G4TPC::ENABLE_STATIC_DISTORTIONS );
175  distortionMap->set_static_distortion_filename( G4TPC::static_distortion_filename );
176 
177  distortionMap->set_do_time_ordered_distortions( G4TPC::ENABLE_TIME_ORDERED_DISTORTIONS );
178  distortionMap->set_time_ordered_distortion_filename( G4TPC::time_ordered_distortion_filename );
179 
180  distortionMap->Init();
181  edrift->setTpcDistortion( distortionMap );
182  }
183 
184  // fudge factors to get drphi 150 microns (in mid and outer Tpc) and dz 500 microns cluster resolution
185  // They represent effects not due to ideal gas properties and ideal readout plane behavior
186  // defaults are 0.085 and 0.105, they can be changed here to get a different resolution
187  edrift->registerPadPlane(padplane);
188  se->registerSubsystem(edrift);
189 
190  // The pad plane readout default is set in PHG4TpcPadPlaneReadout
191  // We may want to change the number of inner layers, and can do that here
192  padplane->set_int_param("tpc_minlayer_inner", G4MVTX::n_maps_layer + G4INTT::n_intt_layer); // sPHENIX layer number of first Tpc readout layer
193  padplane->set_int_param("ntpc_layers_inner", G4TPC::n_tpc_layer_inner);
194  padplane->set_int_param("ntpc_phibins_inner", G4TPC::tpc_layer_rphi_count_inner);
195 
196  // Tpc digitizer
197  //=========
198  PHG4TpcDigitizer* digitpc = new PHG4TpcDigitizer();
199  digitpc->SetTpcMinLayer(G4MVTX::n_maps_layer + G4INTT::n_intt_layer);
200  double ENC = 670.0; // standard
201  digitpc->SetENC(ENC);
202  double ADC_threshold = 4.0 * ENC;
203  digitpc->SetADCThreshold(ADC_threshold); // 4 * ENC seems OK
204  digitpc->Verbosity(verbosity);
205  cout << " Tpc digitizer: Setting ENC to " << ENC << " ADC threshold to " << ADC_threshold
206  << " maps+Intt layers set to " << G4MVTX::n_maps_layer + G4INTT::n_intt_layer << endl;
207 
208  se->registerSubsystem(digitpc);
209 
210 }
