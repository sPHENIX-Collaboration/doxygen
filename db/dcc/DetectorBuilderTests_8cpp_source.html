40  Acts::Experimental::DetectorComponent construct(
41  const Acts::GeometryContext& gctx) const final {
42  auto bounds = std::make_unique<Acts::CuboidVolumeBounds>(10., 10., 10.);
43  // Construct the DetectorVolume
44  auto dVolume =
45  m_sensitives.empty()
46  ? Acts::Experimental::DetectorVolumeFactory::construct(
47  Acts::Experimental::defaultPortalGenerator(), gctx,
48  "TestVolume", Acts::Transform3::Identity(), std::move(bounds),
49  Acts::Experimental::tryAllPortals())
50  : Acts::Experimental::DetectorVolumeFactory::construct(
51  Acts::Experimental::defaultPortalGenerator(), gctx,
52  "TestVolumeWithSurfaces", Acts::Transform3::Identity(),
53  std::move(bounds), m_sensitives, {},
54  Acts::Experimental::tryNoVolumes(),
55  Acts::Experimental::tryAllPortalsAndSurfaces());
56 
57  // Fill the portal container
58  Acts::Experimental::DetectorComponent::PortalContainer portalContainer;
59  for (auto [ip, p] : Acts::enumerate(dVolume->portalPtrs())) {
60  portalContainer[ip] = p;
61  }
62 
63  return Acts::Experimental::DetectorComponent{
64  {dVolume},
65  portalContainer,
66  {{dVolume}, Acts::Experimental::tryRootVolumes()}};
67  }
68 
69  private:
70  std::vector<std::shared_ptr<Acts::Surface>> m_sensitives;
71 };
