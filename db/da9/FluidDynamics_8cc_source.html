29  FluidCellInfo EvolutionHistory::GetAtTimeStep(int id_tau,
30  real x, real y, real eta) {
31  int id_x = GetIdX(x);
32  int id_y = GetIdY(y);
33  int id_eta = GetIdEta(eta);
34  // cijk for idx=i, idy=j and id_eta=k
35  int c000 = CellIndex(id_tau, id_x, id_y, id_eta);
36  int c001 = CellIndex(id_tau, id_x, id_y, id_eta+1);
37  int c010 = CellIndex(id_tau, id_x, id_y+1, id_eta);
38  int c011 = CellIndex(id_tau, id_x, id_y+1, id_eta+1);
39  int c100 = CellIndex(id_tau, id_x+1, id_y, id_eta);
40  int c101 = CellIndex(id_tau, id_x+1, id_y, id_eta+1);
41  int c110 = CellIndex(id_tau, id_x+1, id_y+1, id_eta);
42  int c111 = CellIndex(id_tau, id_x+1, id_y+1, id_eta+1);
43  real x0 = XCoord(id_x);
44  real x1 = XCoord(id_x + 1);
45  real y0 = YCoord(id_y);
46  real y1 = YCoord(id_y + 1);
47  real eta0 = EtaCoord(id_eta);
48  real eta1 = EtaCoord(id_eta + 1);
49 
50  return TrilinearInt(x0, x1, y0, y1, eta0, eta1,
51  data.at(c000), data.at(c001), data.at(c010), data.at(c011),
52  data.at(c100), data.at(c101), data.at(c110), data.at(c111),
53  x, y, eta);
54  }

56  // do interpolation along time direction; we may also need high order
57  // interpolation functions
58  FluidCellInfo EvolutionHistory::get(real tau, real x, real y, real eta){
59  CheckInRange(tau, x, y, eta);
60  int id_tau = GetIdTau(tau);
61  real tau0 = TauCoord(id_tau);
62  real tau1 = TauCoord(id_tau + 1);
63  FluidCellInfo bulk0 = GetAtTimeStep(id_tau, x, y, eta);
64  FluidCellInfo bulk1 = GetAtTimeStep(id_tau+1, x, y, eta);
65  return LinearInt(tau0, tau1, bulk0, bulk1, tau);
66  }
