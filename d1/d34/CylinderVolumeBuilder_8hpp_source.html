102  void midPointAttachZ(VolumeConfig& lConfig) {
103  if (lConfig.zMin >= zMax) {
104  double zMid = 0.5 * (lConfig.zMin + zMax);
105  lConfig.zMin = zMid;
106  zMax = zMid;
107  } else {
108  double zMid = 0.5 * (zMin + lConfig.zMax);
109  lConfig.zMax = zMid;
110  zMin = zMid;
111  }
112  }

118  void attachZ(const VolumeConfig& lConfig) {
119  if (lConfig.zMin >= zMax) {
120  zMax = lConfig.zMin;
121  } else {
122  zMin = lConfig.zMax;
123  }
124  }

220  void configureContainerVolume() {
221  // set the container to be present
222  containerVolumeConfig.present = true;
223  std::string wConditionAddon = "";
224  // if we have more than one config present
225  if ((nVolumeConfig && cVolumeConfig) || (cVolumeConfig && pVolumeConfig) ||
226  (nVolumeConfig && pVolumeConfig)) {
227  wCondition = Wrapping;
228  wConditionScreen = "grouped to ";
229  }
230  // adapt the new volume config to the existing configs
231  if (nVolumeConfig) {
232  containerVolumeConfig.adapt(nVolumeConfig);
233  wConditionScreen += "[n]";
234  }
235  if (cVolumeConfig) {
236  containerVolumeConfig.adapt(cVolumeConfig);
237  wConditionScreen += "[c]";
238  }
239  if (pVolumeConfig) {
240  containerVolumeConfig.adapt(pVolumeConfig);
241  wConditionScreen += "[p]";
242  }
243  // adapt the external one
244  if (externalVolumeConfig) {
245  containerVolumeConfig.adapt(externalVolumeConfig);
246  }
247  // attach the volume configs
248  if (nVolumeConfig && cVolumeConfig) {
249  nVolumeConfig.midPointAttachZ(cVolumeConfig);
250  }
251  if (cVolumeConfig && pVolumeConfig) {
252  cVolumeConfig.midPointAttachZ(pVolumeConfig);
253  }
254  // adapt r afterwards
255  // - easy if no existing volume
256  // - possible if no central volume
257  if (!existingVolumeConfig || !cVolumeConfig) {
258  nVolumeConfig.adaptR(containerVolumeConfig);
259  cVolumeConfig.adaptR(containerVolumeConfig);
260  pVolumeConfig.adaptR(containerVolumeConfig);
261  }
262  }

265  void wrapInsertAttach() {
266  // action is only needed if an existing volume
267  // is present
268  if (existingVolumeConfig) {
269  // 0 - simple attachment case
270  if (!cVolumeConfig) {
271  // check if it can be easily attached
272  if (nVolumeConfig && nVolumeConfig.zMax < existingVolumeConfig.zMin) {
273  nVolumeConfig.attachZ(existingVolumeConfig);
274  // will attach the new volume(s)
275  wCondition = Attaching;
276  wConditionScreen = "[n attached]";
277  }
278  if (pVolumeConfig && pVolumeConfig.zMin > existingVolumeConfig.zMax) {
279  pVolumeConfig.attachZ(existingVolumeConfig);
280  // will attach the new volume(s)
281  wCondition = Attaching;
282  wConditionScreen = "[p attached]";
283  }
284  // see if inner glue volumes are needed
285  if (containerVolumeConfig.rMin > existingVolumeConfig.rMin) {
286  nVolumeConfig.rMin = existingVolumeConfig.rMin;
287  pVolumeConfig.rMin = existingVolumeConfig.rMin;
288  } else {
289  fGapVolumeConfig.present = true;
290  // get the zMin/zMax boundaries
291  fGapVolumeConfig.adaptZ(existingVolumeConfig);
292  fGapVolumeConfig.rMin = containerVolumeConfig.rMin;
293  fGapVolumeConfig.rMax = existingVolumeConfig.rMin;
294  }
295  // see if outer glue volumes are needed
296  if (containerVolumeConfig.rMax < existingVolumeConfig.rMax) {
297  nVolumeConfig.rMax = existingVolumeConfig.rMax;
298  pVolumeConfig.rMax = existingVolumeConfig.rMax;
299  } else {
300  sGapVolumeConfig.present = true;
301  // get the zMin/zMax boundaries
302  sGapVolumeConfig.adaptZ(existingVolumeConfig);
303  sGapVolumeConfig.rMin = existingVolumeConfig.rMax;
304  sGapVolumeConfig.rMax = containerVolumeConfig.rMax;
305  }
306  } else {
307  // full wrapping or full insertion case
308  if (existingVolumeConfig.rMax < containerVolumeConfig.rMin) {
309  // Full wrapping case
310  // - set the rMin
311  nVolumeConfig.rMin = existingVolumeConfig.rMax;
312  cVolumeConfig.rMin = existingVolumeConfig.rMax;
313  pVolumeConfig.rMin = existingVolumeConfig.rMax;
314  // - set the rMax
315  nVolumeConfig.rMax = containerVolumeConfig.rMax;
316  cVolumeConfig.rMax = containerVolumeConfig.rMax;
317  pVolumeConfig.rMax = containerVolumeConfig.rMax;
318  // will wrap the new volume(s) around existing
319  wCondition = Wrapping;
320  wConditionScreen = "[fully wrapped]";
321  } else if (existingVolumeConfig.rMin > containerVolumeConfig.rMax) {
322  // full insertion case
323  // set the rMax
324  nVolumeConfig.rMax = existingVolumeConfig.rMin;
325  cVolumeConfig.rMax = existingVolumeConfig.rMin;
326  pVolumeConfig.rMax = existingVolumeConfig.rMin;
327  // set the rMin
328  nVolumeConfig.rMin = containerVolumeConfig.rMin;
329  cVolumeConfig.rMin = containerVolumeConfig.rMin;
330  pVolumeConfig.rMin = containerVolumeConfig.rMin;
331  // will insert the new volume(s) into existing
332  wCondition = Inserting;
333  wConditionScreen = "[fully inserted]";
334  } else if (cVolumeConfig.wraps(existingVolumeConfig)) {
335  // central wrapping case
336  // set the rMax
337  nVolumeConfig.rMax = containerVolumeConfig.rMax;
338  cVolumeConfig.rMax = containerVolumeConfig.rMax;
339  pVolumeConfig.rMax = containerVolumeConfig.rMax;
340  // set the rMin
341  nVolumeConfig.rMin = existingVolumeConfig.rMin;
342  cVolumeConfig.rMin = existingVolumeConfig.rMax;
343  pVolumeConfig.rMin = existingVolumeConfig.rMin;
344  // set the Central Wrapping
345  wCondition = CentralWrapping;
346  wConditionScreen = "[centrally wrapped]";
347  } else if (existingVolumeConfig.wraps(cVolumeConfig)) {
348  // central insertion case
349  // set the rMax
350  nVolumeConfig.rMax = containerVolumeConfig.rMax;
351  cVolumeConfig.rMax = existingVolumeConfig.rMin;
352  pVolumeConfig.rMax = containerVolumeConfig.rMax;
353  // set the rMin
354  nVolumeConfig.rMin = containerVolumeConfig.rMin;
355  cVolumeConfig.rMin = containerVolumeConfig.rMin;
356  pVolumeConfig.rMin = containerVolumeConfig.rMin;
357  // set the Central Wrapping
358  wCondition = CentralWrapping;
359  wConditionScreen = "[centrally inserted]";
360  } else if ((existingVolumeConfig.rMax > containerVolumeConfig.rMin &&
361  existingVolumeConfig.rMin < containerVolumeConfig.rMin) ||
362  (existingVolumeConfig.rMax > containerVolumeConfig.rMax &&
363  existingVolumeConfig.rMin < containerVolumeConfig.rMax)) {
364  // The volumes are overlapping this shouldn't be happening return an
365  // error
366  throw std::invalid_argument(
367  "Volumes are overlapping, this shouldn't be happening. Please "
368  "check your geometry building.");
369  }
370 
371  // check if gaps are needed
372  //
373  // the gap reference is either the container for FULL wrapping,
374  // insertion
375  // or it is the centralVolume for central wrapping, insertion
376  VolumeConfig referenceVolume =
377  (wCondition == Wrapping || wCondition == Inserting)
378  ? containerVolumeConfig
379  : cVolumeConfig;
380  // - at the negative sector
381  if (existingVolumeConfig.zMin > referenceVolume.zMin) {
382  fGapVolumeConfig.present = true;
383  fGapVolumeConfig.adaptR(existingVolumeConfig);
384  fGapVolumeConfig.zMin = referenceVolume.zMin;
385  fGapVolumeConfig.zMax = existingVolumeConfig.zMin;
386  } else {
387  // adapt lower z boundary
388  if (nVolumeConfig) {
389  nVolumeConfig.zMin = existingVolumeConfig.zMin;
390  } else if (cVolumeConfig) {
391  cVolumeConfig.zMin = existingVolumeConfig.zMin;
392  }
393  }
394  // - at the positive sector
395  if (existingVolumeConfig.zMax < referenceVolume.zMax) {
396  sGapVolumeConfig.present = true;
397  sGapVolumeConfig.adaptR(existingVolumeConfig);
398  sGapVolumeConfig.zMin = existingVolumeConfig.zMax;
399  sGapVolumeConfig.zMax = referenceVolume.zMax;
400  } else {
401  // adapt higher z boundary
402  if (pVolumeConfig) {
403  pVolumeConfig.zMax = existingVolumeConfig.zMax;
404  } else if (cVolumeConfig) {
405  cVolumeConfig.zMax = existingVolumeConfig.zMax;
406  }
407  }
408  }
409  }
410  return;
411  }
