27 Double_t v1_err(Double_t sig, Double_t v1, Double_t Res)
28 {
29  const Double_t Pi = 3.1415927;
30  return 2.3 / sig * sqrt(1 - 16 * v1 * v1 / Pi / Pi) / Res; // 2.3 - coefficient from simple two rapidity bin linear fit
31 }
