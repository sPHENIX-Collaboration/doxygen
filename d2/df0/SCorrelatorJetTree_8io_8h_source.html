154

172 
173  // try to set dca z values

175  m_parSigDcaZ[iParam] = paramDcaZ.at(iParam);
176  } catch (std::out_of_range &out) {

181  assert(paramDcaZ.size() == CONST::NParam);
182  }
183  } // end parameter loop
