282  double z_sum = 0.0;
283  double phi_sum = 0.0;
284  double adc_sum = 0.0;
285  double z2_sum = 0.0;
286  double phi2_sum = 0.0;
287 
288  double radius = layergeom->get_radius(); // returns center of layer
289 
290  int phibinhi = -1;
291  int phibinlo = 666666;
292  int zbinhi = -1;
293  int zbinlo = 666666;
294  int clus_size = ihit_list.size();
295 
296  if(clus_size == 1) return;
297 
298  std::vector<TrkrDefs::hitkey> hitkeyvec;
299  for(auto iter = ihit_list.begin(); iter != ihit_list.end();++iter){
300  double adc = iter->first;
301 
302  if (adc <= 0) continue;
303 
304  int iphi = iter->second.first + phioffset;
305  int iz = iter->second.second + zoffset;
306  if(iphi > phibinhi) phibinhi = iphi;
307  if(iphi < phibinlo) phibinlo = iphi;
308  if(iz > zbinhi) zbinhi = iz;
309  if(iz < zbinlo) zbinlo = iz;
310 
311  // update phi sums
312  double phi_center = layergeom->get_phicenter(iphi);
313  phi_sum += phi_center * adc;
314  phi2_sum += square(phi_center)*adc;
315 
316  // update z sums
317  double z = layergeom->get_zcenter(iz);
318  z_sum += z * adc;
319  z2_sum += square(z)*adc;
320 
321  adc_sum += adc;
322 
323  // capture the hitkeys for all adc values above a certain threshold
324  TrkrDefs::hitkey hitkey = TpcDefs::genHitKey(iphi, iz);
325  // if(adc>5)
326  hitkeyvec.push_back(hitkey);
327  }
328  if (adc_sum < 10) return; // skip obvious noise "clusters"
329 
330  // This is the global position
331  double clusphi = phi_sum / adc_sum;
332  double clusz = z_sum / adc_sum;
333 
334  const double phi_cov = phi2_sum/adc_sum - square(clusphi);
335  const double z_cov = z2_sum/adc_sum - square(clusz);
336 
337  // create the cluster entry directly in the node tree
338

343  // int phi_nsize = phibinhi - phibinlo + 1;
344  // int z_nsize = zbinhi - zbinlo + 1;
345  double phi_size = (double) (phibinhi - phibinlo + 1) * radius * layergeom->get_phistep();
346  double z_size = (double) (zbinhi - zbinlo + 1) * layergeom->get_zstep();
347 
348  // Estimate the errors
349  const double phi_err_square = (phibinhi == phibinlo) ?
350  square(radius*layergeom->get_phistep())/12:
351  square(radius)*phi_cov/(adc_sum*0.14);
352 
353  const double z_err_square = (zbinhi == zbinlo) ?
354  square(layergeom->get_zstep())/12:
355  z_cov/(adc_sum*0.14);
356 
357  // phi_cov = (weighted mean of dphi^2) - (weighted mean of dphi)^2, which is essentially the weighted mean of dphi^2. The error is then:
358  // e_phi = sigma_dphi/sqrt(N) = sqrt( sigma_dphi^2 / N ) -- where N is the number of samples of the distribution with standard deviation sigma_dphi
359  // - N is the number of electrons that drift to the readout plane
360  // We have to convert (sum of adc units for all bins in the cluster) to number of ionization electrons N
361  // Conversion gain is 20 mV/fC - relates total charge collected on pad to PEAK voltage out of ADC. The GEM gain is assumed to be 2000
362  // To get equivalent charge per Z bin, so that summing ADC input voltage over all Z bins returns total input charge, divide voltages by 2.4 for 80 ns SAMPA
363  // Equivalent charge per Z bin is then (ADU x 2200 mV / 1024) / 2.4 x (1/20) fC/mV x (1/1.6e-04) electrons/fC x (1/2000) = ADU x 0.14

365

368  const unsigned int layer = TrkrDefs::getLayer(ckey);
369  const unsigned int side = TpcDefs::getSide(ckey);
370 
371  // correct cluster z for shaping distortion
372  // get parameters of z dependence at this layer
373  double p0, p1;
374  if(clusz < 0)
375  {
376  p0 = par0_neg.first + (double) layer * par0_neg.second;
377  p1= par1_neg.first + (double) layer * par1_neg.second;
378  }
379  else

383  }

385  clusz -= z_correction;
386

388  //================
389  clus->setAdc(adc_sum);
390  clus->setPosition(0, radius * cos(clusphi));
391  clus->setPosition(1, radius * sin(clusphi));
392  clus->setPosition(2, clusz);
393  clus->setGlobal();
394

396  DIM[0][0] = 0.0;
397  DIM[0][1] = 0.0;
398  DIM[0][2] = 0.0;
399  DIM[1][0] = 0.0;

402  DIM[2][0] = 0.0;

546  std::map<TrkrDefs::cluskey, TrkrCluster *> *clusterlist = my_data->clusterlist;
547  std::multimap<TrkrDefs::cluskey, TrkrDefs::hitkey> *clusterhitassoc = my_data->clusterhitassoc;
548 
549  TrkrHitSet *hitset = my_data->hitset;
550  TrkrHitSet::ConstRange hitrangei = hitset->getHits();
551 
552  // for convenience, create a 2D vector to store adc values in and initialize to zero
553  std::vector<std::vector<unsigned short>> adcval(phibins, std::vector<unsigned short>(zbins, 0));
554  std::multimap<unsigned short, ihit> all_hit_map;
555  std::vector<ihit> hit_vect;
556 
557  for (TrkrHitSet::ConstIterator hitr = hitrangei.first;
558  hitr != hitrangei.second;
559  ++hitr){
560  unsigned short phibin = TpcDefs::getPad(hitr->first) - phioffset;
561  unsigned short zbin = TpcDefs::getTBin(hitr->first) - zoffset;
562 
563  float_t fadc = (hitr->second->getAdc()) - pedestal; // proper int rounding +0.5
564  //std::cout << " layer: " << my_data->layer << " phibin " << phibin << " zbin " << zbin << " fadc " << hitr->second->getAdc() << " pedestal " << pedestal << " fadc " << std::endl
565 
566  unsigned short adc = 0;
567  if(fadc>0)
568  adc = (unsigned short) fadc;
569 
570  if(phibin < 0) continue;
571  if(phibin >= phibins) continue;
572  if(zbin < 0) continue;
573  if(zbin >= zbins) continue;
574 
575  if(adc>0){
576  iphiz iCoord(std::make_pair(phibin,zbin));
577  ihit thisHit(adc,iCoord);
578  if(adc>5){
579  all_hit_map.insert(std::make_pair(adc, thisHit));
580  }
581  //adcval[phibin][zbin] = (unsigned short) adc;
582  adcval[phibin][zbin] = (unsigned short) adc;
583  }
584  }
585 
586  int nclus = 0;
587  while(all_hit_map.size()>0){
588 
589  auto iter = all_hit_map.rbegin();
590  if(iter == all_hit_map.rend()){
591  break;
592  }
593  ihit hiHit = iter->second;
594  int iphi = hiHit.second.first;
595  int iz = hiHit.second.second;
596 
597  //put all hits in the all_hit_map (sorted by adc)
598  //start with highest adc hit
599  // -> cluster around it and get vector of hits
600  std::vector<ihit> ihit_list;
601  get_cluster(iphi, iz, phibins, zbins, adcval, ihit_list);
602  nclus++;
603 
604  // -> calculate cluster parameters
605  // -> add hits to truth association
606  // remove hits from all_hit_map
607  // repeat untill all_hit_map empty
608  calc_cluster_parameter(ihit_list,nclus++, layergeom, hitset,phioffset,zoffset, par0_neg, par0_pos, par1_neg, par1_pos, clusterlist, clusterhitassoc, do_assoc,tGeometry, surfMaps);
609  remove_hits(ihit_list,all_hit_map, adcval);
610  }
611  pthread_exit(nullptr);
612 }

779  // The hits are stored in hitsets, where each hitset contains all hits in a given TPC readout (layer, sector, side), so clusters are confined to a hitset
780  // The TPC clustering is more complicated than for the silicon, because we have to deal with overlapping clusters
781 
782  // loop over the TPC HitSet objects
783  TrkrHitSetContainer::ConstRange hitsetrange = m_hits->getHitSets(TrkrDefs::TrkrId::tpcId);
784  const int num_hitsets = std::distance(hitsetrange.first,hitsetrange.second);
785 
786  // create structure to store given thread and associated data
787  struct thread_pair_t
788  {
789  pthread_t thread;
790  thread_data data;
791  };
792 
793  // create vector of thread pairs and reserve the right size upfront to avoid reallocation
794  std::vector<thread_pair_t> threads;
795  threads.reserve( num_hitsets );
796

800
