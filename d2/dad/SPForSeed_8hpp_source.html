145 template <typename SpacePoint>
146 inline void SPForSeed<SpacePoint>::set(SpacePoint* const& sp, const float* r) {
147  spacepoint = sp;
148  m_x = r[0];
149  m_y = r[1];
150  m_z = r[2];
151  m_r = std::hypot(m_x, m_y);
152  m_surface = sp->surface;
153  m_q = 100000.;
154 
155  if (!sp->clusterList().second) {
156  m_covr = sp->covr * 9.;
157  m_covz = sp->covz * 9.;
158  if (m_covr < 0.06) {
159  m_covr = 0.06;
160  }
161  if (m_covz < 0.06) {
162  m_covz = 0.06;
163  }
164  } else {
165  m_covr = sp->covr * 8.;
166  m_covz = sp->covz * 8.;
167  if (m_covr < 0.1) {
168  m_covr = 0.1;
169  }
170  if (m_covz < 0.1) {
171  m_covz = 0.1;
172  }
173  }
174 }
